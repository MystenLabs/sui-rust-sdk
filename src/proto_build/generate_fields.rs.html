12pub(crate) fn generate_field_info(packages: &HashMap<String, FileDescriptorSet>, out_dir: &Path) {
13    for (package, fds) in packages {
14        if package.contains("google") {
15            continue;
16        }
17
18        let mut buf = String::new();
19        let mut stream = TokenStream::new();
20
21        for file in &fds.file {
22            for message in &file.message_type {
23                stream.extend(generate_field_info_for_message(message));
24            }
25        }
26
27        let code = quote! {
28            mod _field_impls {
29                use super::*;
30
31                use crate::field::MessageFields;
32                use crate::field::MessageField;
33
34                #stream
35            }
36        };
37
38        let ast: syn::File = syn::parse2(code).expect("not a valid tokenstream");
39        let code = prettyplease::unparse(&ast);
40        buf.push_str(&code);
41
42        let file_name = format!("{package}.field_info.rs");
43        std::fs::write(out_dir.join(file_name), &buf).unwrap();
44    }
45}

47fn generate_field_info_for_message(message: &DescriptorProto) -> TokenStream {
48    let map_types: HashSet<String> = message
49        .nested_type
50        .iter()
51        .filter_map(|m| {
52            if m.options.as_ref().is_some_and(|o| o.map_entry()) {
53                Some(m.name().to_owned())
54            } else {
55                None
56            }
57        })
58        .collect();
59
60    let constants = generate_field_constants(message, &map_types);
61    let message_fields_impl = generate_message_fields_impl(message);
62
63    quote! {
64        #constants
65        #message_fields_impl
66    }
67}

69fn generate_field_constants(message: &DescriptorProto, map_types: &HashSet<String>) -> TokenStream {
70    let message_ident = quote::format_ident!("{}", message.name());
71    let mut field_consts = TokenStream::new();
72
73    for field in &message.field {
74        field_consts.extend(generate_field_constant(message.name(), field, map_types));
75    }
76
77    quote! {
78        impl #message_ident {
79            #field_consts
80        }
81    }
82}

84fn generate_message_fields_impl(message: &DescriptorProto) -> TokenStream {
85    let message_ident = quote::format_ident!("{}", message.name());
86
87    let mut field_refs = TokenStream::new();
88
89    for field in &message.field {
90        field_refs.extend(generate_field_reference(field));
91    }
92
93    quote! {
94        impl MessageFields for #message_ident {
95            const FIELDS: &'static [&'static MessageField] = &[
96                #field_refs
97            ];
98        }
99    }
100}

102fn generate_field_constant(
103    message_name: &str,
104    field: &FieldDescriptorProto,
105    map_types: &HashSet<String>,
106) -> TokenStream {
107    let ident = quote::format_ident!("{}_FIELD", field.name().to_ascii_uppercase());
108    let name = field.name();
109    let json_name = field.json_name();
110    let number = field.number();
111
112    let message_fields =
113        if matches!(field.r#type(), Type::Message) && !field.type_name().contains("google") {
114            let field_message_name = field.type_name().split('.').next_back().unwrap();
115
116            if field_message_name == message_name || map_types.contains(field_message_name) {
117                quote! { None }
118            } else {
119                let field_message = quote::format_ident!("{}", field_message_name);
120                quote! { Some(#field_message::FIELDS) }
121            }
122        } else {
123            quote! { None }
124        };
125
126    quote! {
127        pub const #ident: &'static MessageField = &MessageField {
128            name: #name,
129            json_name: #json_name,
130            number: #number,
131            message_fields: #message_fields,
132        };
133    }
134}

136fn generate_field_reference(field: &FieldDescriptorProto) -> TokenStream {
137    let ident = quote::format_ident!("{}_FIELD", field.name().to_ascii_uppercase());
138
139    quote! {
140        Self::#ident,
141    }
