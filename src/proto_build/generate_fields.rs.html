12pub(crate) fn generate_field_info(packages: &HashMap<String, FileDescriptorSet>, out_dir: &Path) {
13    for (package, fds) in packages {
14        if package.contains("google") {
15            continue;
16        }
17
18        let mut buf = String::new();
19        let mut stream = TokenStream::new();
20
21        for file in &fds.file {
22            for message in &file.message_type {
23                stream.extend(generate_field_info_for_message(message));
24            }
25        }
26
27        let code = quote! {
28            mod _field_impls {
29                #![allow(clippy::wrong_self_convention)]
30
31                use super::*;
32
33                use crate::field::MessageFields;
34                use crate::field::MessageField;
35
36                #stream
37            }
38        };
39
40        let ast: syn::File = syn::parse2(code).expect("not a valid tokenstream");
41        let code = prettyplease::unparse(&ast);
42        buf.push_str(&code);
43
44        let file_name = format!("{package}.field_info.rs");
45        std::fs::write(out_dir.join(file_name), &buf).unwrap();
46    }
47}

49fn generate_field_info_for_message(message: &DescriptorProto) -> TokenStream {
50    let map_types: HashSet<String> = message
51        .nested_type
52        .iter()
53        .filter_map(|m| {
54            if m.options.as_ref().is_some_and(|o| o.map_entry()) {
55                Some(m.name().to_owned())
56            } else {
57                None
58            }
59        })
60        .collect();
61
62    let constants = generate_field_constants(message, &map_types);
63    let message_fields_impl = generate_message_fields_impl(message);
64    let field_path_builders = generate_field_path_builders_impl(message, &map_types);
65
66    quote! {
67        #constants
68        #message_fields_impl
69        #field_path_builders
70    }
71}

73fn generate_field_constants(message: &DescriptorProto, map_types: &HashSet<String>) -> TokenStream {
74    let message_ident = quote::format_ident!("{}", message.name());
75    let mut field_consts = TokenStream::new();
76
77    for field in &message.field {
78        field_consts.extend(generate_field_constant(message.name(), field, map_types));
79    }
80
81    quote! {
82        impl #message_ident {
83            #field_consts
84        }
85    }
86}

88fn generate_message_fields_impl(message: &DescriptorProto) -> TokenStream {
89    let message_ident = quote::format_ident!("{}", message.name());
90
91    let mut field_refs = TokenStream::new();
92
93    for field in &message.field {
94        field_refs.extend(generate_field_reference(field));
95    }
96
97    quote! {
98        impl MessageFields for #message_ident {
99            const FIELDS: &'static [&'static MessageField] = &[
100                #field_refs
101            ];
102        }
103    }
104}

106fn generate_field_constant(
107    message_name: &str,
108    field: &FieldDescriptorProto,
109    map_types: &HashSet<String>,
110) -> TokenStream {
111    let ident = quote::format_ident!("{}_FIELD", field.name().to_ascii_uppercase());
112    let name = field.name();
113    let json_name = field.json_name();
114    let number = field.number();
115
116    let message_fields =
117        if matches!(field.r#type(), Type::Message) && !field.type_name().contains("google") {
118            let field_message_name = field.type_name().split('.').next_back().unwrap();
119
120            if field_message_name == message_name || map_types.contains(field_message_name) {
121                quote! { None }
122            } else {
123                let field_message = quote::format_ident!("{}", field_message_name);
124                quote! { Some(#field_message::FIELDS) }
125            }
126        } else {
127            quote! { None }
128        };
129
130    quote! {
131        pub const #ident: &'static MessageField = &MessageField {
132            name: #name,
133            json_name: #json_name,
134            number: #number,
135            message_fields: #message_fields,
136        };
137    }
138}

140fn generate_field_reference(field: &FieldDescriptorProto) -> TokenStream {
141    let ident = quote::format_ident!("{}_FIELD", field.name().to_ascii_uppercase());
142
143    quote! {
144        Self::#ident,
145    }

148fn generate_field_path_builders_impl(
149    message: &DescriptorProto,
150    map_types: &HashSet<String>,
151) -> TokenStream {
152    let message_ident = quote::format_ident!("{}", message.name());
153    let builder_ident = quote::format_ident!("{}FieldPathBuilder", message.name());
154
155    let mut field_chain_methods = TokenStream::new();
156
157    for field in &message.field {
158        field_chain_methods.extend(generate_field_chain_methods(
159            message.name(),
160            field,
161            map_types,
162        ));
163    }
164
165    quote! {
166        impl #message_ident {
167            pub fn path_builder() -> #builder_ident {
168                #builder_ident::new()
169            }
170        }
171
172        pub struct #builder_ident {
173            path: Vec<&'static str>,
174        }
175
176        impl #builder_ident {
177            #[allow(clippy::new_without_default)]
178            pub fn new() -> Self {
179                Self {
180                    path: Default::default(),
181                }
182            }
183
184            #[doc(hidden)]
185            pub fn new_with_base(base: Vec<&'static str>) -> Self {
186                Self { path: base }
187            }
188
189            pub fn finish(self) -> String {
190                self.path.join(".")
191            }
192
193            #field_chain_methods
194        }
195    }
196}

198fn generate_field_chain_methods(
199    message_name: &str,
200    field: &FieldDescriptorProto,
201    map_types: &HashSet<String>,
202) -> TokenStream {
203    let message_ident = quote::format_ident!("{message_name}");
204    let field_const = quote::format_ident!("{}_FIELD", field.name().to_ascii_uppercase());
205    let name = if field.name() == "type" {
206        quote::format_ident!("r#{}", field.name())
207    } else {
208        quote::format_ident!("{}", field.name())
209    };
210
211    if matches!(field.r#type(), Type::Message) && !field.type_name().contains("google") {
212        let field_message_name = field.type_name().split('.').next_back().unwrap();
213
214        if field_message_name == message_name || map_types.contains(field_message_name) {
215            quote! {
216                pub fn #name(mut self) -> String {
217                    self.path.push(#message_ident::#field_const.name);
218                    self.finish()
219                }
220            }
221        } else {
222            let return_type = quote::format_ident!("{}FieldPathBuilder", field_message_name);
223            quote! {
224                pub fn #name(mut self) -> #return_type {
225                    self.path.push(#message_ident::#field_const.name);
226                    #return_type::new_with_base(self.path)
227                }
228            }
229        }
230    } else {
231        quote! {
232            pub fn #name(mut self) -> String {
233                self.path.push(#message_ident::#field_const.name);
234                self.finish()
235            }
236        }
237    }
