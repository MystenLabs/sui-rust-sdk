9fn validate_proto_path(path: &str) -> Result<(), String> {
10    if path.chars().next().map(|c| c != '.').unwrap_or(true) {
11        return Err(format!(
12            "Protobuf paths must be fully qualified (begin with a leading '.'): {path}",
13        ));
14    }
15    if path.split('.').skip(1).any(str::is_empty) {
16        return Err(format!("invalid fully-qualified Protobuf path: {path}"));
17    }
18    Ok(())
19}

21#[derive(Debug)]
22pub struct ExternPaths {
23    extern_paths: HashMap<String, String>,
24}
25
26impl ExternPaths {
27    pub fn new(paths: &[(String, String)], prost_types: bool) -> Result<ExternPaths, String> {
28        let mut extern_paths = ExternPaths {
29            extern_paths: HashMap::new(),
30        };
31
32        for (proto_path, rust_path) in paths {
33            extern_paths.insert(proto_path.clone(), rust_path.clone())?;
34        }
35
36        if prost_types {
37            extern_paths.insert(".google.protobuf".to_string(), "::prost_types".to_string())?;
38            extern_paths.insert(".google.protobuf.BoolValue".to_string(), "bool".to_string())?;
39            extern_paths.insert(
40                ".google.protobuf.BytesValue".to_string(),
41                "::alloc::vec::Vec<u8>".to_string(),
42            )?;
43            extern_paths.insert(
44                ".google.protobuf.DoubleValue".to_string(),
45                "f64".to_string(),
46            )?;
47            extern_paths.insert(".google.protobuf.Empty".to_string(), "()".to_string())?;
48            extern_paths.insert(".google.protobuf.FloatValue".to_string(), "f32".to_string())?;
49            extern_paths.insert(".google.protobuf.Int32Value".to_string(), "i32".to_string())?;
50            extern_paths.insert(".google.protobuf.Int64Value".to_string(), "i64".to_string())?;
51            extern_paths.insert(
52                ".google.protobuf.StringValue".to_string(),
53                "::alloc::string::String".to_string(),
54            )?;
55            extern_paths.insert(
56                ".google.protobuf.UInt32Value".to_string(),
57                "u32".to_string(),
58            )?;
59            extern_paths.insert(
60                ".google.protobuf.UInt64Value".to_string(),
61                "u64".to_string(),
62            )?;
63        }
64
65        Ok(extern_paths)
66    }
67
68    fn insert(&mut self, proto_path: String, rust_path: String) -> Result<(), String> {
69        validate_proto_path(&proto_path)?;
70        match self.extern_paths.entry(proto_path) {
71            hash_map::Entry::Occupied(occupied) => {
72                return Err(format!(
73                    "duplicate extern Protobuf path: {}",
74                    occupied.key()
75                ));
76            }
77            hash_map::Entry::Vacant(vacant) => vacant.insert(rust_path),
78        };
79        Ok(())
80    }
81
82    pub fn resolve_ident(&self, pb_ident: &str) -> Option<String> {
83        // protoc should always give fully qualified identifiers.
84        assert_eq!(".", &pb_ident[..1]);
85
86        if let Some(rust_path) = self.extern_paths.get(pb_ident) {
87            return Some(rust_path.clone());
88        }
89
90        // TODO(danburkert): there must be a more efficient way to do this, maybe a trie?
91        for (idx, _) in pb_ident.rmatch_indices('.') {
92            if let Some(rust_path) = self.extern_paths.get(&pb_ident[..idx]) {
93                let mut segments = pb_ident[idx + 1..].split('.');
94                let ident_type = segments.next_back().map(to_upper_camel);
95
96                return Some(
97                    rust_path
98                        .split("::")
99                        .chain(segments)
100                        .enumerate()
101                        .map(|(idx, segment)| {
102                            if idx == 0 && segment == "crate" {
103                                // If the first segment of the path is 'crate', then do not escape
104                                // it into a raw identifier, since it's being used as the keyword.
105                                segment.to_owned()
106                            } else {
107                                to_snake(segment)
108                            }
109                        })
110                        .chain(ident_type.into_iter())
111                        .join("::"),
112                );
113            }
114        }
115
116        None
117    }
118}

125    #[test]
126    fn test_extern_paths() {
127        let paths = ExternPaths::new(
128            &[
129                (".foo".to_string(), "::foo1".to_string()),
130                (".foo.bar".to_string(), "::foo2".to_string()),
131                (".foo.baz".to_string(), "::foo3".to_string()),
132                (".foo.Fuzz".to_string(), "::foo4::Fuzz".to_string()),
133                (".a.b.c.d.e.f".to_string(), "::abc::def".to_string()),
134            ],
135            false,
136        )
137        .unwrap();
138
139        let case = |proto_ident: &str, resolved_ident: &str| {
140            assert_eq!(paths.resolve_ident(proto_ident).unwrap(), resolved_ident);
141        };
142
143        case(".foo", "::foo1");
144        case(".foo.Foo", "::foo1::Foo");
145        case(".foo.bar", "::foo2");
146        case(".foo.Bas", "::foo1::Bas");
147
148        case(".foo.bar.Bar", "::foo2::Bar");
149        case(".foo.Fuzz.Bar", "::foo4::fuzz::Bar");
150
151        case(".a.b.c.d.e.f", "::abc::def");
152        case(".a.b.c.d.e.f.g.FooBar.Baz", "::abc::def::g::foo_bar::Baz");
153
154        assert!(paths.resolve_ident(".a").is_none());
155        assert!(paths.resolve_ident(".a.b").is_none());
156        assert!(paths.resolve_ident(".a.c").is_none());
157    }
