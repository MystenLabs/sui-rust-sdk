14impl Context {
15    pub fn new(extern_paths: extern_paths::ExternPaths, graph: DescriptorGraph) -> Self {
16        Self {
17            extern_paths,
18            graph,
19        }
20    }
21
22    // Used to check if a path is external and should be skipped
23    pub fn is_extern(&self, fq_proto_name: &str) -> bool {
24        self.extern_paths.resolve_ident(fq_proto_name).is_some()
25    }
26
27    // Resolves protobuf type to a rust path relative to package/scope.
28    pub fn resolve_ident(&self, package: &str, pb_ident: &str) -> String {
29        // protoc should always give fully qualified identifiers.
30        assert_eq!(".", &pb_ident[..1]);
31
32        if let Some(proto_ident) = self.extern_paths.resolve_ident(pb_ident) {
33            return proto_ident;
34        }
35
36        let mut local_path = package.split('.').peekable();
37
38        // If no package is specified the start of the package name will be '.'
39        // and split will return an empty string ("") which breaks resolution
40        // The fix to this is to ignore the first item if it is empty.
41        if local_path.peek().is_some_and(|s| s.is_empty()) {
42            local_path.next();
43        }
44
45        let mut ident_path = pb_ident[1..].split('.');
46        let ident_type = ident_path.next_back().unwrap();
47        let mut ident_path = ident_path.peekable();
48
49        // Skip path elements in common.
50        while local_path.peek().is_some() && local_path.peek() == ident_path.peek() {
51            local_path.next();
52            ident_path.next();
53        }
54
55        local_path
56            .map(|_| "super".to_string())
57            .chain(ident_path.map(to_snake))
58            .chain(std::iter::once(to_upper_camel(ident_type)))
59            .join("::")
60    }
61
62    pub fn graph(&self) -> &DescriptorGraph {
63        &self.graph
64    }
