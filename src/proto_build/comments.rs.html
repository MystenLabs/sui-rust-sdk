1use prost_types::source_code_info::Location;

3/// Comments on a Protobuf item.
4#[derive(Debug, Default, Clone)]
5#[allow(unused)]
6pub struct Comments {
7    /// Leading detached blocks of comments.
8    pub leading_detached: Vec<Vec<String>>,
9
10    /// Leading comments.
11    pub leading: Vec<String>,
12
13    /// Trailing comments.
14    pub trailing: Vec<String>,
15}
16
17impl Comments {
18    pub(crate) fn from_location(location: &Location) -> Comments {
19        let leading_detached = location
20            .leading_detached_comments
21            .iter()
22            .map(get_lines)
23            .collect();
24        let leading = location
25            .leading_comments
26            .as_ref()
27            .map_or(Vec::new(), get_lines);
28        let trailing = location
29            .trailing_comments
30            .as_ref()
31            .map_or(Vec::new(), get_lines);
32        Comments {
33            leading_detached,
34            leading,
35            trailing,
36        }
37    }
38}
39
40fn get_lines<S>(comments: S) -> Vec<String>
41where
42    S: AsRef<str>,
43{
44    comments.as_ref().lines().map(str::to_owned).collect()
45}
