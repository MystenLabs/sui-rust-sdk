26/// `MessageGraph` builds a graph of messages whose edges correspond to nesting.
27/// The goal is to recognize when message types are recursively nested, so
28/// that fields can be boxed when necessary.
29pub struct DescriptorGraph {
30    index: BTreeMap<String, NodeIndex>,
31    graph: Graph<String, ()>,
32
33    pub packages: BTreeSet<String>,
34    pub messages: BTreeMap<String, Message>,
35    enums: BTreeMap<String, Enum>,
36    services: BTreeMap<String, Service>,
37}
38
39impl DescriptorGraph {
40    pub(crate) fn new<'a>(files: impl Iterator<Item = &'a FileDescriptorProto>) -> DescriptorGraph {
41        let mut graph = DescriptorGraph {
42            index: BTreeMap::new(),
43            graph: Graph::new(),
44            messages: BTreeMap::new(),
45            enums: BTreeMap::new(),
46            services: BTreeMap::new(),
47            packages: BTreeSet::new(),
48        };
49
50        for file in files {
51            let package = format!(
52                "{}{}",
53                if file.package.is_some() { "." } else { "" },
54                file.package.as_deref().unwrap_or("")
55            );
56            for msg in &file.message_type {
57                graph.add_message_edges(&package, msg);
58            }
59
60            graph.packages.insert(package);
61            FileParser::parse(&mut graph, file.clone());
62        }
63
64        graph
65    }
66
67    fn get_or_insert_index(&mut self, msg_name: String) -> NodeIndex {
68        assert_eq!(b'.', msg_name.as_bytes()[0]);
69        *self
70            .index
71            .entry(msg_name.clone())
72            .or_insert_with(|| self.graph.add_node(msg_name))
73    }
74
75    /// Adds message to graph IFF it contains a non-repeated field containing another message.
76    /// The purpose of the message graph is detecting recursively nested messages and co-recursively nested messages.
77    /// Because prost does not box message fields, recursively nested messages would not compile in Rust.
78    /// To allow recursive messages, the message graph is used to detect recursion and automatically box the recursive field.
79    /// Since repeated messages are already put in a Vec, boxing them isn’t necessary even if the reference is recursive.
80    fn add_message_edges(&mut self, package: &str, msg: &DescriptorProto) {
81        let msg_name = format!("{}.{}", package, msg.name.as_ref().unwrap());
82        let msg_index = self.get_or_insert_index(msg_name.clone());
83
84        for field in &msg.field {
85            if field.r#type() == Type::Message && field.label() != Label::Repeated {
86                let field_index = self.get_or_insert_index(field.type_name.clone().unwrap());
87                self.graph.add_edge(msg_index, field_index, ());
88            }
89        }
90
91        for msg in &msg.nested_type {
92            self.add_message_edges(&msg_name, msg);
93        }
94    }
95
96    /// Returns true if message type `inner` is nested in message type `outer`.
97    #[allow(unused)]
98    pub fn is_nested(&self, outer: &str, inner: &str) -> bool {
99        let outer = match self.index.get(outer) {
100            Some(outer) => *outer,
101            None => return false,
102        };
103        let inner = match self.index.get(inner) {
104            Some(inner) => *inner,
105            None => return false,
106        };
107
108        has_path_connecting(&self.graph, outer, inner, None)
109    }
110}

112struct FileParser<'a> {
113    graph: &'a mut DescriptorGraph,
114    package: String,
115    type_path: Vec<String>,
116    source_info: Option<SourceCodeInfo>,
117    path: Vec<i32>,
118}
119
120impl<'a> FileParser<'a> {
121    fn parse(graph: &'a mut DescriptorGraph, file: FileDescriptorProto) {
122        let source_info = file.source_code_info.map(|mut s| {
123            s.location.retain(|loc| {
124                let len = loc.path.len();
125                len > 0 && len % 2 == 0
126            });
127            s.location.sort_by(|a, b| a.path.cmp(&b.path));
128            s
129        });
130
131        let package = format!(
132            "{}{}",
133            if file.package.is_some() { "." } else { "" },
134            file.package.as_deref().unwrap_or("")
135        );
136
137        let mut parser = Self {
138            graph,
139            package,
140            type_path: Vec::new(),
141            path: Vec::new(),
142            source_info,
143        };
144
145        // Messages
146        parser.path.push(4);
147        for (idx, message) in file.message_type.into_iter().enumerate() {
148            parser.path.push(idx as i32);
149            parser.process_message(message);
150            parser.path.pop();
151        }
152        parser.path.pop();
153
154        // Enums
155        parser.path.push(5);
156        for (idx, desc) in file.enum_type.into_iter().enumerate() {
157            parser.path.push(idx as i32);
158            parser.process_enum(desc);
159            parser.path.pop();
160        }
161        parser.path.pop();
162
163        // Services
164        parser.path.push(6);
165        for (idx, service) in file.service.into_iter().enumerate() {
166            parser.path.push(idx as i32);
167            parser.process_service(service);
168            parser.path.pop();
169        }
170
171        parser.path.pop();
172    }
173
174    fn process_service(&mut self, service: ServiceDescriptorProto) {
175        let name = service.name().to_owned();
176        // debug!("  service: {:?}", name);
177        let comments = self.comments();
178        self.path.push(2);
179        let methods = service
180            .method
181            .into_iter()
182            .enumerate()
183            .map(|(idx, mut method)| {
184                // debug!("  method: {:?}", method.name());
185
186                self.path.push(idx as i32);
187                let comments = self.comments();
188                self.path.pop();
189
190                let name = method.name.take().unwrap();
191                let input_proto_type = method.input_type.take().unwrap();
192                let output_proto_type = method.output_type.take().unwrap();
193                //TODO FIX
194                // let input_type = self.resolve_ident(&input_proto_type);
195                // let output_type = self.resolve_ident(&output_proto_type);
196                let input_type = String::new();
197                let output_type = String::new();
198                let client_streaming = method.client_streaming();
199                let server_streaming = method.server_streaming();
200
201                Method {
202                    name: to_snake(&name),
203                    proto_name: name,
204                    comments,
205                    input_type,
206                    output_type,
207                    input_proto_type,
208                    output_proto_type,
209                    options: method.options.unwrap_or_default(),
210                    client_streaming,
211                    server_streaming,
212                }
213            })
214            .collect();
215        self.path.pop();
216
217        let service = Service {
218            name: to_upper_camel(&name),
219            proto_name: name,
220            package: self.package.clone(),
221            comments,
222            methods,
223            options: service.options.unwrap_or_default(),
224        };
225        let full_name = format!("{}{}", service.package, service.proto_name);
226        self.graph.services.insert(full_name, service);
227    }
228
229    fn process_message(&mut self, descriptor: DescriptorProto) {
230        let type_name = self.fq_name(descriptor.name());
231
232        // Split the nested message types into a vector of normal nested message types, and a map
233        // of the map field entry types. The path index of the nested message types is preserved so
234        // that comments can be retrieved.
235        type NestedTypes = Vec<(DescriptorProto, usize)>;
236        type MapTypes = BTreeMap<String, (FieldDescriptorProto, FieldDescriptorProto)>;
237        let (nested_types, map_types): (NestedTypes, MapTypes) = descriptor
238            .nested_type
239            .iter()
240            .enumerate()
241            .partition_map(|(idx, nested_type)| {
242                if nested_type
243                    .options
244                    .as_ref()
245                    .and_then(|options| options.map_entry)
246                    .unwrap_or(false)
247                {
248                    let key = nested_type.field[0].clone();
249                    let value = nested_type.field[1].clone();
250                    assert_eq!("key", key.name());
251                    assert_eq!("value", value.name());
252
253                    let name = format!("{}.{}", &type_name, nested_type.name());
254                    Either::Right((name, (key, value)))
255                } else {
256                    Either::Left((nested_type.clone(), idx))
257                }
258            });
259
260        // Split the fields into a vector of the normal fields, and oneof fields.
261        // Path indexes are preserved so that comments can be retrieved.
262        let mut fields: Vec<Field> = Vec::new();
263        let mut oneof_map: BTreeMap<i32, Vec<Field>> = Default::default();
264        self.path.push(2);
265        for (idx, proto) in descriptor.field.iter().enumerate() {
266            self.path.push(idx as i32);
267            let comments = self.comments();
268            let map = map_types.get(proto.type_name()).cloned();
269
270            if let Some(oneof_index) = proto.oneof_index
271                && !proto.proto3_optional()
272            {
273                // oneof
274                oneof_map.entry(oneof_index).or_default().push(Field {
275                    inner: proto.clone(),
276                    map,
277                    comments,
278                });
279            } else {
280                // normal field
281                fields.push(Field {
282                    inner: proto.clone(),
283                    map,
284                    comments,
285                });
286            }
287            self.path.pop();
288        }
289        self.path.pop();
290
291        self.path.push(8);
292        let oneof_fields: Vec<OneofField> = descriptor
293            .oneof_decl
294            .iter()
295            .enumerate()
296            .filter_map(|(idx, proto)| {
297                let idx = idx as i32;
298                self.path.push(idx);
299                let oneof = if let Some(fields) = oneof_map.remove(&idx) {
300                    let comments = self.comments();
301                    Some(OneofField::new(
302                        &descriptor,
303                        proto.clone(),
304                        fields,
305                        comments,
306                    ))
307                } else {
308                    None
309                };
310                self.path.pop();
311                oneof
312            })
313            .collect();
314        self.path.pop();
315
316        // Handle Nested Messagse and Enums
317        self.type_path.push(descriptor.name().to_owned());
318
319        self.path.push(3);
320        for (nested_type, idx) in nested_types {
321            self.path.push(idx as i32);
322            self.process_message(nested_type);
323            self.path.pop();
324        }
325        self.path.pop();
326
327        self.path.push(4);
328        for (idx, nested_enum) in descriptor.enum_type.clone().into_iter().enumerate() {
329            self.path.push(idx as i32);
330            self.process_enum(nested_enum);
331            self.path.pop();
332        }
333        self.path.pop();
334
335        self.type_path.pop();
336        // End Handling Nested Messagse and Enums
337
338        let message = Message {
339            descriptor,
340            package: self.package.clone(),
341            type_name,
342            fields,
343            oneof_fields,
344            comments: self.comments(),
345        };
346        self.graph
347            .messages
348            .insert(message.type_name.clone(), message);
349    }
350
351    fn process_enum(&mut self, desc: EnumDescriptorProto) {
352        let type_name = self.fq_name(desc.name());
353        let mut values = Vec::new();
354
355        self.path.push(2);
356        for (idx, value) in desc.value.clone().into_iter().enumerate() {
357            self.path.push(idx as i32);
358            values.push(EnumValue {
359                inner: value,
360                comments: self.comments(),
361            });
362            self.path.pop();
363        }
364        self.path.pop();
365
366        let e = Enum {
367            type_name,
368            package: self.package.clone(),
369            inner: desc,
370            comments: self.comments(),
371            values,
372        };
373        self.graph.enums.insert(e.type_name.clone(), e);
374    }
375
376    fn comments_opt(&self) -> Option<Comments> {
377        let source_info = self.source_info.as_ref()?;
378        let idx = source_info
379            .location
380            .binary_search_by_key(&self.path.as_slice(), |location| &location.path[..])
381            .unwrap();
382        let location = source_info.location.get(idx)?;
383        Some(Comments::from_location(location))
384    }
385
386    fn comments(&self) -> Comments {
387        self.comments_opt().unwrap_or_default()
388    }
389
390    /// Returns the fully-qualified name, starting with a dot
391    fn fq_name(&self, message_name: &str) -> String {
392        format!(
393            "{}{}{}{}.{}",
394            if self.package.is_empty() { "" } else { "." },
395            self.package.trim_matches('.'),
396            if self.type_path.is_empty() { "" } else { "." },
397            self.type_path.join("."),
398            message_name,
399        )
400    }
401}

431impl Field {
432    pub fn rust_struct_field_name(&self) -> String {
433        crate::ident::sanitize_identifier(self.inner.name())
434    }
435
436    pub fn is_optional(&self) -> bool {
437        if self.inner.proto3_optional.unwrap_or(false) {
438            return true;
439        }
440
441        if self.inner.label() != Label::Optional {
442            return false;
443        }
444
445        match self.inner.r#type() {
446            Type::Message => true,
447            _ => false,
448            // _ => self.syntax == Syntax::Proto2,
449        }
450    }
451
452    pub fn is_map(&self) -> bool {
453        self.map.is_some()
454    }
455
456    pub fn is_repeated(&self) -> bool {
457        self.inner.label() == Label::Repeated
458    }
459
460    pub fn resolve_rust_type_path(
461        &self,
462        context: &crate::context::Context,
463        package: &str,
464    ) -> String {
465        match self.inner.r#type() {
466            Type::Float => String::from("f32"),
467            Type::Double => String::from("f64"),
468            Type::Uint32 | Type::Fixed32 => String::from("u32"),
469            Type::Uint64 | Type::Fixed64 => String::from("u64"),
470            Type::Int32 | Type::Sfixed32 | Type::Sint32 => String::from("i32"),
471            Type::Int64 | Type::Sfixed64 | Type::Sint64 => String::from("i64"),
472            Type::Bool => String::from("bool"),
473            Type::String => String::from("String"),
474            Type::Bytes => String::from("::prost::bytes::Bytes"),
475            Type::Group | Type::Message | Type::Enum => {
476                context.resolve_ident(package, self.inner.type_name())
477            }
478        }
479    }
480
481    pub fn is_message(&self) -> bool {
482        matches!(self.inner.r#type(), Type::Message)
483    }
484
485    pub fn is_well_known_type(&self) -> bool {
486        self.inner.type_name().starts_with(".google.protobuf")
487    }
488
489    pub fn is_enum(&self) -> bool {
490        matches!(self.inner.r#type(), Type::Enum)
491    }
492}

496pub struct OneofField {
497    pub descriptor: OneofDescriptorProto,
498    pub fields: Vec<Field>,
499
500    pub comments: Comments,
501
502    // This type has the same name as another nested type at the same level
503    pub has_type_name_conflict: bool,
504}
505
506impl OneofField {
507    fn new(
508        parent: &DescriptorProto,
509        descriptor: OneofDescriptorProto,
510        fields: Vec<Field>,
511        comments: Comments,
512    ) -> Self {
513        let has_type_name_conflict = parent
514            .nested_type
515            .iter()
516            .any(|nested| to_snake(nested.name()) == descriptor.name());
517
518        Self {
519            descriptor,
520            fields,
521            comments,
522            has_type_name_conflict,
523        }
524    }
525
526    pub fn rust_struct_field_name(&self) -> String {
527        crate::ident::sanitize_identifier(self.descriptor.name())
528    }
