1use prost_types::FileDescriptorSet;

15fn main() {
16    let root_dir = PathBuf::from(std::env!("CARGO_MANIFEST_DIR"));
17    let proto_dir = root_dir
18        .join("../sui-rpc/vendored/proto")
19        .canonicalize()
20        .unwrap();
21    let out_dir = root_dir
22        .join("../sui-rpc/src/proto/generated")
23        .canonicalize()
24        .unwrap();
25
26    let proto_ext = std::ffi::OsStr::new("proto");
27    let proto_files = walkdir::WalkDir::new(&proto_dir)
28        .into_iter()
29        .filter_map(|entry| {
30            (|| {
31                let entry = entry?;
32                if entry.file_type().is_dir() {
33                    return Ok(None);
34                }
35
36                let path = entry.into_path();
37                if path.extension() != Some(proto_ext) {
38                    return Ok(None);
39                }
40
41                Ok(Some(path))
42            })()
43            .transpose()
44        })
45        .collect::<Result<Vec<_>, walkdir::Error>>()
46        .unwrap();
47
48    let mut fds = protox::Compiler::new(std::slice::from_ref(&proto_dir))
49        .unwrap()
50        .include_source_info(true)
51        .include_imports(true)
52        .open_files(&proto_files)
53        .unwrap()
54        .file_descriptor_set();
55    // Sort files by name to have deterministic codegen output
56    fds.file.sort_by(|a, b| a.name.cmp(&b.name));
57
58    if let Err(error) = tonic_prost_build::configure()
59        .build_client(true)
60        .build_server(true)
61        .bytes(".")
62        .boxed(".sui.rpc.v2.Input.literal")
63        .boxed(".sui.rpc.v2.Epoch.system_state")
64        .boxed("json")
65        .message_attribute(".sui.rpc", "#[non_exhaustive]")
66        .enum_attribute(".sui.rpc", "#[non_exhaustive]")
67        .btree_map(".")
68        .out_dir(&out_dir)
69        .compile_fds(fds.clone())
70    {
71        panic!("failed to compile protos: {error}");
72    }
73
74    // Setup for extended codegen
75    let extern_paths = context::extern_paths::ExternPaths::new(&[], true).unwrap();
76    let graph = DescriptorGraph::new(fds.file.iter());
77    let context = context::Context::new(extern_paths, graph);
78    codegen::accessors::generate_accessors(&context, &out_dir);
79
80    // Group the files by their package, in order to have a single fds file per package, and have
81    // the files in the package sorted by their filename in order have a stable serialized format.
82    let mut packages: HashMap<_, FileDescriptorSet> = HashMap::new();
83    for mut file in fds.file {
84        // Clear out the source code info as its not required for reflection
85        file.source_code_info = None;
86        packages
87            .entry(file.package().to_owned())
88            .or_default()
89            .file
90            .push(file);
91    }
92
93    generate_fields::generate_field_info(&packages, &out_dir);
94
95    let mut json_builder = pbjson_build::Builder::new();
96
97    for file in packages.values().flat_map(|set| set.file.iter()) {
98        json_builder.register_file_descriptor(file.to_owned());
99    }
100
101    json_builder
102        .out_dir(&out_dir)
103        .ignore_unknown_fields()
104        .btree_map(["."])
105        .build(&[".google.rpc", ".sui"])
106        .unwrap();
107
108    for (package, fds) in packages {
109        let file_name = format!("{package}.fds.bin");
110        let file_descriptor_set_path = out_dir.join(&file_name);
111        std::fs::write(file_descriptor_set_path, fds.encode_to_vec()).unwrap();
112    }
113
114    let status = std::process::Command::new("git")
115        .arg("diff")
116        .arg("--exit-code")
117        .arg("--")
118        .arg(out_dir)
119        .status();
120    match status {
121        Ok(status) if !status.success() => panic!("You should commit the protobuf files"),
122        Err(error) => panic!("failed to run `git diff`: {error}"),
123        Ok(_) => {}
124    }
125}
