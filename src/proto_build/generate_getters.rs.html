16pub(crate) fn generate_getters(packages: &HashMap<String, FileDescriptorSet>, out_dir: &Path) {
17    for (package, fds) in packages {
18        if package.contains("google") {
19            continue;
20        }
21
22        let mut buf = String::new();
23        let mut stream = TokenStream::new();
24
25        for file in &fds.file {
26            for message in &file.message_type {
27                stream.extend(generate_getters_for_message(message));
28            }
29        }
30
31        let code = quote! {
32            mod _getter_impls {
33                #![allow(clippy::useless_conversion)]
34                use super::*;
35
36                #stream
37            }
38        };
39
40        let ast: syn::File = syn::parse2(code).expect("not a valid tokenstream");
41        let code = prettyplease::unparse(&ast);
42        buf.push_str(&code);
43
44        let file_name = format!("{package}.getters.rs");
45        std::fs::write(out_dir.join(file_name), &buf).unwrap();
46    }
47}

49fn generate_getters_for_message(message: &DescriptorProto) -> TokenStream {
50    let map_types: HashSet<String> = message
51        .nested_type
52        .iter()
53        .filter_map(|m| {
54            if m.options.as_ref().is_some_and(|o| o.map_entry()) {
55                Some(m.name().to_owned())
56            } else {
57                None
58            }
59        })
60        .collect();
61
62    let functions = generate_getter_functions(message, &map_types);
63
64    quote! {
65        #functions
66    }
67}

69fn generate_getter_functions(
70    message: &DescriptorProto,
71    map_types: &HashSet<String>,
72) -> TokenStream {
73    let message_ident = quote::format_ident!("{}", message.name());
74    let mut functions = TokenStream::new();
75    let mut const_default_fields = TokenStream::new();
76
77    let mut oneofs = BTreeSet::new();
78
79    for field in &message.field {
80        if let Some(function) = generate_getter_function(message, field, map_types) {
81            functions.extend(function);
82        }
83
84        if let Some(const_default_field) =
85            generate_const_default_field(field, map_types, &mut oneofs)
86        {
87            const_default_fields.extend(const_default_field);
88        }
89    }
90
91    for oneof_index in oneofs {
92        let oneof_field = quote::format_ident!("{}", message.oneof_decl[oneof_index].name());
93        const_default_fields.extend(quote! {
94            #oneof_field: None,
95        });
96    }
97
98    quote! {
99        impl #message_ident {
100            pub const fn const_default() -> Self {
101                Self {
102                    #const_default_fields
103                }
104            }
105
106            #[doc(hidden)]
107            pub fn default_instance() -> &'static Self {
108                static DEFAULT: #message_ident = #message_ident::const_default();
109                &DEFAULT
110            }
111
112            #functions
113        }
114    }
115}

117fn generate_getter_function(
118    message: &DescriptorProto,
119    field: &FieldDescriptorProto,
120    map_types: &HashSet<String>,
121) -> Option<TokenStream> {
122    let name = quote::format_ident!(
123        "{}{}",
124        if field.name() == "type" { "r#" } else { "" },
125        field.name()
126    );
127    let name_opt = quote::format_ident!("{}_opt", field.name());
128    let name_mut = quote::format_ident!("{}_mut", field.name());
129    let name_opt_mut = quote::format_ident!("{}_opt_mut", field.name());
130    let with_name = quote::format_ident!("with_{}", field.name());
131
132    // Skip google messages
133    if field.type_name().contains("google") {
134        return None;
135    }
136
137    let field_message_name = resolve_type(field);
138    let field_message_ty = TokenStream::from_str(&field_message_name).unwrap(); // quote::format_ident!("{}", field_message_name);
139
140    // Skip maps for now
141    if map_types.contains(&field_message_name) {
142        return None;
143    }
144
145    // oneofs
146    let function = if matches!(field.r#type(), Type::Enum) {
147        // skip Enums
148        return None;
149    } else if field.oneof_index.is_some() && !field.proto3_optional() {
150        let default_instance = TokenStream::from_str(&type_default(field)).unwrap();
151        let ref_return_type = TokenStream::from_str(&getter_ref_return_type(field)).unwrap();
152        let oneof = &message.oneof_decl[field.oneof_index() as usize];
153        let oneof_field = quote::format_ident!("{}", oneof.name());
154        let oneof_name = quote::format_ident!("{}", oneof.name().to_pascal_case());
155        let message_mod = quote::format_ident!("{}", message.name().to_snake_case());
156        let variant = quote::format_ident!("{}", field.name().to_pascal_case());
157        let field_as = if is_ref_return(field) {
158            quote! {field as _}
159        } else {
160            quote! {*field}
161        };
162
163        quote! {
164            pub fn #name(&self) -> #ref_return_type {
165                if let Some(#message_mod::#oneof_name::#variant(field)) = &self.#oneof_field {
166                    #field_as
167                    // field as _
168                } else {
169                    // #field_message_ty::default_instance() as _
170                    #default_instance
171                }
172            }
173
174            pub fn #name_opt(&self) -> Option<#ref_return_type> {
175                if let Some(#message_mod::#oneof_name::#variant(field)) = &self.#oneof_field {
176                    Some(#field_as)
177                } else {
178                    None
179                }
180            }
181
182            pub fn #name_opt_mut(&mut self) -> Option<&mut #field_message_ty> {
183                if let Some(#message_mod::#oneof_name::#variant(field)) = &mut self.#oneof_field {
184                    Some(field as _)
185                } else {
186                    None
187                }
188            }
189
190            pub fn #name_mut(&mut self) -> &mut #field_message_ty {
191                if self.#name_opt_mut().is_none() {
192                    self.#oneof_field = Some(#message_mod::#oneof_name::#variant(#field_message_ty::default()));
193                }
194                self.#name_opt_mut().unwrap()
195            }
196
197            pub fn #with_name(mut self, field: #field_message_ty) -> Self {
198                self.#oneof_field = Some(#message_mod::#oneof_name::#variant(field.into()));
199                self
200            }
201        }
202    } else if matches!(
203        field.label(),
204        prost_types::field_descriptor_proto::Label::Repeated
205    ) {
206        quote! {
207            pub fn #name(&self) -> &[#field_message_ty] {
208                &self.#name
209            }
210
211            pub fn #with_name(mut self, field: Vec<#field_message_ty>) -> Self {
212                self.#name = field;
213                self
214            }
215        }
216    } else if !matches!(field.r#type(), Type::Message) {
217        //primitives
218        quote! {
219            // pub fn #name_opt(&self) -> Option<&#field_message_ty> {
220            //     self.#name
221            //         .as_ref()
222            //         .map(|field| field as _)
223            // }
224
225            // pub fn #name_opt_mut(&mut self) -> Option<&mut #field_message_ty> {
226            //     self.#name
227            //         .as_mut()
228            //         .map(|field| field as _)
229            // }
230
231            // pub fn #name_mut(&mut self) -> &mut #field_message_ty {
232            //     self.#name
233            //         .get_or_insert_default()
234            // }
235
236            pub fn #with_name(mut self, field: #field_message_ty) -> Self {
237                self.#name = Some(field.into());
238                self
239            }
240        }
241    } else {
242        quote! {
243            pub fn #name(&self) -> &#field_message_ty {
244                self.#name
245                    .as_ref()
246                    .map(|field| field as _)
247                    .unwrap_or_else(|| #field_message_ty::default_instance() as _)
248            }
249
250            pub fn #name_opt(&self) -> Option<&#field_message_ty> {
251                self.#name
252                    .as_ref()
253                    .map(|field| field as _)
254            }
255
256            pub fn #name_opt_mut(&mut self) -> Option<&mut #field_message_ty> {
257                self.#name
258                    .as_mut()
259                    .map(|field| field as _)
260            }
261
262            pub fn #name_mut(&mut self) -> &mut #field_message_ty {
263                self.#name
264                    .get_or_insert_default()
265            }
266
267            pub fn #with_name(mut self, field: #field_message_ty) -> Self {
268                self.#name = Some(field.into());
269                self
270            }
271        }
272    };
273
274    Some(function)
275}

277fn generate_const_default_field(
278    field: &FieldDescriptorProto,
279    map_types: &HashSet<String>,
280    oneofs: &mut BTreeSet<usize>,
281) -> Option<TokenStream> {
282    let name = quote::format_ident!(
283        "{}{}",
284        if field.name() == "type" { "r#" } else { "" },
285        field.name()
286    );
287    let is_map = if matches!(field.r#type(), Type::Message) && !field.type_name().contains("google")
288    {
289        let field_message_name = field.type_name().split('.').next_back().unwrap();
290        map_types.contains(field_message_name)
291    } else {
292        false
293    };
294
295    // Skip oneofs that are not synthetic optional
296    if field.oneof_index.is_some() && !field.proto3_optional() {
297        oneofs.insert(field.oneof_index() as usize);
298        return None;
299    }
300
301    let const_default_field = if is_map {
302        // Map fields
303        quote! {
304            #name: std::collections::BTreeMap::new(),
305        }
306    } else if matches!(
307        field.label(),
308        prost_types::field_descriptor_proto::Label::Repeated
309    ) {
310        // Repeated fields
311        quote! {
312            #name: Vec::new(),
313        }
314    } else {
315        quote! {
316            #name: None,
317        }
318    };
319
320    Some(const_default_field)
321}

323fn resolve_type(field: &FieldDescriptorProto) -> String {
324    match field.r#type() {
325        Type::Float => String::from("f32"),
326        Type::Double => String::from("f64"),
327        Type::Uint32 | Type::Fixed32 => String::from("u32"),
328        Type::Uint64 | Type::Fixed64 => String::from("u64"),
329        Type::Int32 | Type::Sfixed32 | Type::Sint32 | Type::Enum => String::from("i32"),
330        Type::Int64 | Type::Sfixed64 | Type::Sint64 => String::from("i64"),
331        Type::Bool => String::from("bool"),
332        Type::String => String::from("String"),
333        Type::Bytes => String::from("::prost::bytes::Bytes"),
334        Type::Group | Type::Message => field.type_name().split('.').next_back().unwrap().into(),
335    }
336}

338fn type_default(field: &FieldDescriptorProto) -> String {
339    match field.r#type() {
340        Type::Float => String::from("0.0f32"),
341        Type::Double => String::from("0.0f64"),
342        Type::Uint32 | Type::Fixed32 => String::from("0u32"),
343        Type::Uint64 | Type::Fixed64 => String::from("0u64"),
344        Type::Int32 | Type::Sfixed32 | Type::Sint32 | Type::Enum => String::from("0i32"),
345        Type::Int64 | Type::Sfixed64 | Type::Sint64 => String::from("0i64"),
346        Type::Bool => String::from("false"),
347        Type::String => String::from("\"\""),
348        Type::Bytes => String::from("&[]"),
349        Type::Group | Type::Message => {
350            let ty = field.type_name().split('.').next_back().unwrap();
351            format!("{}::default_instance() as _", ty)
352        }
353    }
354}

356fn getter_ref_return_type(field: &FieldDescriptorProto) -> String {
357    match field.r#type() {
358        Type::Float => String::from("f32"),
359        Type::Double => String::from("f64"),
360        Type::Uint32 | Type::Fixed32 => String::from("u32"),
361        Type::Uint64 | Type::Fixed64 => String::from("u64"),
362        Type::Int32 | Type::Sfixed32 | Type::Sint32 | Type::Enum => String::from("i32"),
363        Type::Int64 | Type::Sfixed64 | Type::Sint64 => String::from("i64"),
364        Type::Bool => String::from("bool"),
365        Type::String => String::from("&str"),
366        Type::Bytes => String::from("&[u8]"),
367        Type::Group | Type::Message => {
368            let ty = field.type_name().split('.').next_back().unwrap();
369            format!("&{}", ty)
370        }
371    }
372}

374fn is_ref_return(field: &FieldDescriptorProto) -> bool {
375    match field.r#type() {
376        Type::Float => false,
377        Type::Double => false,
378        Type::Uint32 | Type::Fixed32 => false,
379        Type::Uint64 | Type::Fixed64 => false,
380        Type::Int32 | Type::Sfixed32 | Type::Sint32 | Type::Enum => false,
381        Type::Int64 | Type::Sfixed64 | Type::Sint64 => false,
382        Type::Bool => false,
383        Type::String => true,
384        Type::Bytes => true,
385        Type::Group | Type::Message => true,
386    }
