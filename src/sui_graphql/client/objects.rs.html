15impl Client {
16    /// Fetch an object by its ID and deserialize from BCS.
17    ///
18    /// Returns:
19    /// - `Ok(Some(object))` if the object exists
20    /// - `Ok(None)` if the object does not exist
21    /// - `Err(Error::Request)` for network errors
22    /// - `Err(Error::Base64)` / `Err(Error::Bcs)` for decoding errors
23    ///
24    /// # Example
25    ///
26    /// ```no_run
27    /// # async fn example() -> Result<(), Box<dyn std::error::Error>> {
28    /// use sui_graphql::Client;
29    /// use sui_sdk_types::Address;
30    ///
31    /// let client = Client::new("https://sui-mainnet.mystenlabs.com/graphql")?;
32    /// let object_id: Address = "0x5".parse()?;
33    ///
34    /// match client.get_object(object_id).await? {
35    ///     Some(object) => println!("Object version: {}", object.version()),
36    ///     None => println!("Object not found"),
37    /// }
38    /// # Ok(())
39    /// # }
40    /// ```
41    pub async fn get_object(&self, object_id: Address) -> Result<Option<Object>, Error> {
42        #[derive(Response)]
43        struct Response {
44            #[field(path = "object.objectBcs")]
45            object: Option<Bcs<Object>>,
46        }
47
48        const QUERY: &str = r#"
49            query($id: SuiAddress!) {
50                object(address: $id) {
51                    objectBcs
52                }
53            }
54        "#;
55
56        let variables = serde_json::json!({ "id": object_id });
57
58        let response = self.query::<Response>(QUERY, variables).await?;
59
60        Ok(response.into_data().and_then(|d| d.object).map(|b| b.0))
61    }

63    /// Stream all objects owned by an address.
64    ///
65    /// Handles pagination automatically, fetching pages as needed.
66    ///
67    /// # Example
68    ///
69    /// ```no_run
70    /// # async fn example() -> Result<(), Box<dyn std::error::Error>> {
71    /// use futures::StreamExt;
72    /// use std::pin::pin;
73    /// use sui_graphql::Client;
74    /// use sui_sdk_types::Address;
75    ///
76    /// let client = Client::new("https://sui-testnet.mystenlabs.com/graphql")?;
77    /// let owner: Address = "0x123...".parse()?;
78    ///
79    /// let mut stream = pin!(client.list_objects(owner));
80    /// while let Some(result) = stream.next().await {
81    ///     let object = result?;
82    ///     println!("Object version: {}", object.version());
83    /// }
84    /// # Ok(())
85    /// # }
86    /// ```
87    pub fn list_objects(&self, owner: Address) -> impl Stream<Item = Result<Object, Error>> + '_ {
88        let client = self.clone();
89        paginate(move |cursor| {
90            let client = client.clone();
91            async move { client.fetch_objects_page(owner, cursor.as_deref()).await }
92        })
93    }
94
95    /// Fetch a single page of objects owned by an address.
96    async fn fetch_objects_page(
97        &self,
98        owner: Address,
99        cursor: Option<&str>,
100    ) -> Result<Page<Object>, Error> {
101        #[derive(Response)]
102        struct Response {
103            #[field(path = "objects.pageInfo")]
104            page_info: Option<PageInfo>,
105            #[field(path = "objects.nodes[].objectBcs")]
106            objects: Option<Vec<Bcs<Object>>>,
107        }
108
109        const QUERY: &str = r#"
110            query($owner: SuiAddress!, $after: String) {
111                objects(filter: { owner: $owner }, after: $after) {
112                    pageInfo {
113                        hasNextPage
114                        endCursor
115                    }
116                    nodes {
117                        objectBcs
118                    }
119                }
120            }
121        "#;
122
123        let variables = serde_json::json!({
124            "owner": owner,
125            "after": cursor,
126        });
127
128        let response = self.query::<Response>(QUERY, variables).await?;
129
130        let data = response.into_data();
131        let page_info = data
132            .as_ref()
133            .and_then(|d| d.page_info.clone())
134            .unwrap_or(PageInfo {
135                has_next_page: false,
136                end_cursor: None,
137            });
138
139        let objects = data
140            .and_then(|d| d.objects)
141            .unwrap_or_default()
142            .into_iter()
143            .map(|b| b.0)
144            .collect();
145
146        Ok(Page {
147            items: objects,
148            has_next_page: page_info.has_next_page,
149            end_cursor: page_info.end_cursor,
150        })
151    }
