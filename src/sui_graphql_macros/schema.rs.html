1//! GraphQL schema parsing and type lookup.
2
3use graphql_parser::schema::Definition;
4use graphql_parser::schema::TypeDefinition;
5use graphql_parser::schema::{self as gql};
6use std::collections::HashMap;
7use std::sync::LazyLock;
8
9/// The embedded Sui GraphQL schema.
10const SCHEMA_SDL: &str = include_str!("../schema/sui.graphql");
11
12/// Parsed and indexed schema, cached for reuse across macro invocations.
13static SCHEMA: LazyLock<Result<Schema, String>> =
14    LazyLock::new(|| Schema::parse(SCHEMA_SDL).map_err(|e| format!("Failed to parse schema: {e}")));
15
16/// A parsed GraphQL schema with type lookup.
17#[derive(Debug)]
18pub struct Schema {
19    types: HashMap<String, TypeInfo>,
20}
21
22/// Information about a GraphQL type.
23#[derive(Debug, Clone)]
24pub struct TypeInfo {
25    pub name: String,
26    pub fields: HashMap<String, FieldInfo>,
27}
28
29/// Information about a field on a type.
30#[derive(Debug, Clone)]
31pub struct FieldInfo {
32    pub name: String,
33    pub type_name: String,
34    pub is_list: bool,
35}
36
37impl Schema {
38    /// Load the embedded schema.
39    pub fn load() -> Result<&'static Schema, syn::Error> {
40        SCHEMA
41            .as_ref()
42            .map_err(|e| syn::Error::new(proc_macro2::Span::call_site(), e.clone()))
43    }
44
45    /// Parse a GraphQL SDL schema.
46    fn parse(sdl: &str) -> Result<Self, graphql_parser::schema::ParseError> {
47        let doc = gql::parse_schema::<String>(sdl)?;
48        let mut types = HashMap::new();
49
50        for def in doc.definitions {
51            if let Definition::TypeDefinition(type_def) = def {
52                let type_info = Self::parse_type_definition(type_def);
53                types.insert(type_info.name.clone(), type_info);
54            }
55        }
56
57        Ok(Schema { types })
58    }
59
60    fn parse_type_definition(def: TypeDefinition<String>) -> TypeInfo {
61        match def {
62            TypeDefinition::Object(obj) => TypeInfo {
63                name: obj.name.clone(),
64                fields: obj
65                    .fields
66                    .iter()
67                    .map(|f| {
68                        let info = Self::parse_field(f);
69                        (info.name.clone(), info)
70                    })
71                    .collect(),
72            },
73            TypeDefinition::Interface(i) => TypeInfo {
74                name: i.name.clone(),
75                fields: i
76                    .fields
77                    .iter()
78                    .map(|f| {
79                        let info = Self::parse_field(f);
80                        (info.name.clone(), info)
81                    })
82                    .collect(),
83            },
84            TypeDefinition::Scalar(s) => TypeInfo {
85                name: s.name.clone(),
86                fields: HashMap::new(),
87            },
88            TypeDefinition::Enum(e) => TypeInfo {
89                name: e.name.clone(),
90                fields: HashMap::new(),
91            },
92            TypeDefinition::InputObject(io) => TypeInfo {
93                name: io.name.clone(),
94                fields: HashMap::new(),
95            },
96            TypeDefinition::Union(u) => TypeInfo {
97                name: u.name.clone(),
98                fields: HashMap::new(),
99            },
100        }
101    }
102
103    fn parse_field(field: &gql::Field<String>) -> FieldInfo {
104        let (type_name, is_list) = Self::parse_type(&field.field_type);
105        FieldInfo {
106            name: field.name.clone(),
107            type_name,
108            is_list,
109        }
110    }
111
112    /// Parse a GraphQL type, extracting the base type name and list status.
113    fn parse_type(ty: &gql::Type<String>) -> (String, bool) {
114        match ty {
115            gql::Type::NamedType(name) => (name.clone(), false),
116            gql::Type::NonNullType(inner) => Self::parse_type(inner),
117            gql::Type::ListType(inner) => {
118                let (name, _) = Self::parse_type(inner);
119                (name, true)
120            }
121        }
122    }
123
124    /// Look up a field on a type.
125    pub fn get_field(&self, type_name: &str, field_name: &str) -> Option<&FieldInfo> {
126        self.types.get(type_name)?.fields.get(field_name)
127    }
128
129    /// Get all field names for a type.
130    pub fn field_names(&self, type_name: &str) -> Vec<&str> {
131        self.types
132            .get(type_name)
133            .map(|t| t.fields.keys().map(|s| s.as_str()).collect())
134            .unwrap_or_default()
135    }
136}
