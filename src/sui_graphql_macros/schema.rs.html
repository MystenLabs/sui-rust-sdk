1//! GraphQL schema parsing and type lookup.
2
3use graphql_parser::schema as gql;
4use graphql_parser::schema::Definition;
5use graphql_parser::schema::TypeDefinition;
6use std::collections::HashMap;
7use std::sync::LazyLock;
8
9/// The embedded Sui GraphQL schema.
10const SCHEMA_SDL: &str = include_str!("../schema/sui.graphql");
11
12/// Parsed and indexed schema, cached for reuse across macro invocations.
13static SCHEMA: LazyLock<Result<Schema, String>> =
14    LazyLock::new(|| Schema::parse(SCHEMA_SDL).map_err(|e| format!("Failed to parse schema: {e}")));
15
16/// A parsed GraphQL schema with type lookup.
17#[derive(Debug)]
18pub struct Schema {
19    types: HashMap<String, TypeInfo>,
20}
21
22/// Information about a GraphQL type.
23#[derive(Debug)]
24pub struct TypeInfo {
25    pub name: String,
26    pub fields: HashMap<String, FieldInfo>,
27}
28
29/// Information about a field on a type.
30#[derive(Debug)]
31pub struct FieldInfo {
32    pub name: String,
33    pub type_name: String,
34    pub is_list: bool,
35}
36
37impl Schema {
38    /// Load the embedded Sui GraphQL schema.
39    ///
40    /// Returns an `Arc<Schema>` for cheap sharing without deep copying.
41    pub fn load() -> Result<&'static Schema, syn::Error> {
42        SCHEMA
43            .as_ref()
44            .map_err(|e| syn::Error::new(proc_macro2::Span::call_site(), e.clone()))
45    }
46
47    /// Load a custom schema from SDL content.
48    ///
49    /// Used when `#[response(schema = "...")]` specifies a custom schema.
50    pub fn from_sdl(sdl: &str) -> Result<Self, syn::Error> {
51        Self::parse(sdl).map_err(|e| {
52            syn::Error::new(
53                proc_macro2::Span::call_site(),
54                format!("Failed to parse custom schema: {e}"),
55            )
56        })
57    }
58
59    /// Parse a GraphQL SDL schema.
60    fn parse(sdl: &str) -> Result<Self, graphql_parser::schema::ParseError> {
61        let doc = gql::parse_schema::<String>(sdl)?;
62        let mut types = HashMap::new();
63
64        for def in doc.definitions {
65            if let Definition::TypeDefinition(type_def) = def {
66                let type_info = Self::parse_type_definition(type_def);
67                types.insert(type_info.name.clone(), type_info);
68            }
69        }
70
71        Ok(Schema { types })
72    }
73
74    fn parse_type_definition(def: TypeDefinition<String>) -> TypeInfo {
75        match def {
76            TypeDefinition::Object(obj) => TypeInfo {
77                name: obj.name.clone(),
78                fields: obj
79                    .fields
80                    .iter()
81                    .map(|f| {
82                        let info = Self::parse_field(f);
83                        (info.name.clone(), info)
84                    })
85                    .collect(),
86            },
87            TypeDefinition::Interface(i) => TypeInfo {
88                name: i.name.clone(),
89                fields: i
90                    .fields
91                    .iter()
92                    .map(|f| {
93                        let info = Self::parse_field(f);
94                        (info.name.clone(), info)
95                    })
96                    .collect(),
97            },
98            TypeDefinition::Scalar(s) => TypeInfo {
99                name: s.name.clone(),
100                fields: HashMap::new(),
101            },
102            TypeDefinition::Enum(e) => TypeInfo {
103                name: e.name.clone(),
104                fields: HashMap::new(),
105            },
106            TypeDefinition::InputObject(io) => TypeInfo {
107                name: io.name.clone(),
108                fields: HashMap::new(),
109            },
110            TypeDefinition::Union(u) => TypeInfo {
111                name: u.name.clone(),
112                fields: HashMap::new(),
113            },
114        }
115    }
116
117    fn parse_field(field: &gql::Field<String>) -> FieldInfo {
118        let (type_name, is_list) = Self::parse_type(&field.field_type);
119        FieldInfo {
120            name: field.name.clone(),
121            type_name,
122            is_list,
123        }
124    }
125
126    /// Parse a GraphQL type, extracting the base type name and list status.
127    fn parse_type(ty: &gql::Type<String>) -> (String, bool) {
128        match ty {
129            gql::Type::NamedType(name) => (name.clone(), false),
130            gql::Type::NonNullType(inner) => Self::parse_type(inner),
131            gql::Type::ListType(inner) => {
132                let (name, _) = Self::parse_type(inner);
133                (name, true)
134            }
135        }
136    }
137
138    /// Look up a field on a type.
139    pub fn get_field(&self, type_name: &str, field_name: &str) -> Option<&FieldInfo> {
140        self.types.get(type_name)?.fields.get(field_name)
141    }
142
143    /// Get all field names for a type.
144    pub fn field_names(&self, type_name: &str) -> Vec<&str> {
145        self.types
146            .get(type_name)
147            .map(|t| t.fields.keys().map(|s| s.as_str()).collect())
148            .unwrap_or_default()
149    }
150}
