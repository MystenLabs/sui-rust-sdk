1//! Procedural macros for sui-graphql with compile-time schema validation.
2//!
3//! This crate provides the `Response` derive macro that:
4//! - Validates field paths against the Sui GraphQL schema at compile time
5//! - Generates deserialization code for extracting nested fields from JSON responses
6//!
7//! # Example
8//!
9//! ```ignore
10//! use sui_graphql_macros::Response;
11//!
12//! #[derive(Response)]
13//! struct ObjectData {
14//!     #[field(path = "object.address")]
15//!     address: String,
16//!     #[field(path = "object.version")]
17//!     version: u64,
18//! }
19//!
20//! // The macro validates paths against the schema and generates extraction code.
21//! // Invalid paths like "object.nonexistent" will cause a compile error.
22//! ```
23

39/// Derive macro for GraphQL response types with nested field extraction.
40///
41/// Use `#[field(path = "...")]` to specify the JSON path to extract each field.
42/// Paths are dot-separated (e.g., `"object.address"` extracts `json["object"]["address"]`).
43///
44/// # Generated Code
45///
46/// The macro generates:
47/// - `from_value(serde_json::Value) -> Result<Self, String>` method
48/// - `Deserialize` implementation that uses `from_value`
49///
50/// # Example
51///
52/// ```ignore
53/// #[derive(Response)]
54/// struct ChainInfo {
55///     #[field(path = "chainIdentifier")]
56///     chain_id: String,
57///
58///     #[field(path = "epoch.epochId")]
59///     epoch_id: Option<u64>,
60/// }
61/// ```
62#[proc_macro_derive(Response, attributes(response, field))]
63pub fn derive_query_response(input: TokenStream) -> TokenStream {
64    let input = parse_macro_input!(input as DeriveInput);
65
66    match derive_query_response_impl(input) {
67        Ok(tokens) => tokens.into(),
68        Err(err) => err.to_compile_error().into(),
69    }
70}
71
72fn derive_query_response_impl(input: DeriveInput) -> Result<TokenStream2, syn::Error> {
73    // Darling input structures - define the "schema" for macro input.
74    // Darling generates parsing code automatically, including error messages for invalid input.
75
76    #[derive(Debug, FromDeriveInput)]
77    #[darling(attributes(response), supports(struct_named))]
78    struct ResponseInput {
79        ident: syn::Ident,                           // Struct name
80        generics: syn::Generics,                     // Generic parameters
81        data: darling::ast::Data<(), ResponseField>, // Struct fields
82        #[darling(default)]
83        schema: Option<String>, // Custom schema path: #[response(schema = "path/to/schema.graphql")]
84    }
85
86    #[derive(Debug, FromField)]
87    #[darling(attributes(field))]
88    struct ResponseField {
89        ident: Option<syn::Ident>,
90        ty: syn::Type,
91        path: SpannedValue<String>, // Required - darling will error if missing
92        #[darling(default)]
93        skip_schema_validation: bool,
94    }
95
96    let parsed = ResponseInput::from_derive_input(&input)?;
97
98    // Load the GraphQL schema for validation.
99    // If a custom schema path is provided, load it; otherwise use the embedded Sui schema.
100    let loaded_schema = if let Some(path) = &parsed.schema {
101        // Resolve path relative to the crate's directory.
102        // SUI_GRAPHQL_SCHEMA_DIR is used by trybuild tests (which run from a temp directory).
103        let base_dir = std::env::var("SUI_GRAPHQL_SCHEMA_DIR")
104            .or_else(|_| std::env::var("CARGO_MANIFEST_DIR"))
105            .unwrap();
106        let full_path = std::path::Path::new(&base_dir).join(path);
107        let sdl = std::fs::read_to_string(&full_path).map_err(|e| {
108            syn::Error::new(
109                proc_macro2::Span::call_site(),
110                format!(
111                    "Failed to read schema from '{}': {}",
112                    full_path.display(),
113                    e
114                ),
115            )
116        })?;
117        Some(schema::Schema::from_sdl(&sdl)?)
118    } else {
119        None
120    };
121    let schema = if let Some(schema) = &loaded_schema {
122        schema
123    } else {
124        schema::Schema::load()?
125    };
126
127    let fields = parsed
128        .data
129        .as_ref()
130        .take_struct()
131        .ok_or_else(|| syn::Error::new_spanned(&input, "Response only supports structs"))?
132        .fields;
133
134    // Generate extraction code for each field
135    let mut field_extractions = Vec::new();
136    let mut field_names = Vec::new();
137
138    for field in &fields {
139        let field_ident = field
140            .ident
141            .as_ref()
142            .ok_or_else(|| syn::Error::new_spanned(&input, "Unnamed fields not supported"))?;
143
144        // Parse path once - used for both validation and code generation
145        let spanned_path = &field.path;
146        let parsed_path = path::ParsedPath::parse(spanned_path.as_str())
147            .map_err(|e| syn::Error::new(spanned_path.span(), e.to_string()))?;
148
149        // Validate path against GraphQL schema
150        let terminal_type = if !field.skip_schema_validation {
151            Some(validation::validate_path_against_schema(
152                schema,
153                &parsed_path,
154                spanned_path.span(),
155            )?)
156        } else {
157            None
158        };
159
160        // Skip Vec excess check when schema validation is skipped (user takes full
161        // responsibility) or when the terminal type is an object-like scalar (e.g., JSON)
162        // whose value can be an array.
163        let skip_vec_excess_check = field.skip_schema_validation
164            || terminal_type.is_some_and(validation::is_object_like_scalar);
165        validation::validate_type_matches_path(&parsed_path, &field.ty, skip_vec_excess_check)?;
166
167        // Generate extraction code using the same parsed path
168        let type_structure = validation::analyze_type(&field.ty);
169        let extraction = generate_field_extraction(&parsed_path, &type_structure, field_ident);
170        field_extractions.push(extraction);
171        field_names.push(field_ident);
172    }
173
174    let struct_ident = &parsed.ident;
175    let (impl_generics, ty_generics, where_clause) = parsed.generics.split_for_impl();
176
177    // Generate both `from_value` and `Deserialize` impl:
178    //
179    // - `from_value`: Core extraction logic, parses from serde_json::Value
180    // - `Deserialize`: Allows direct use with serde (e.g., `serde_json::from_str::<MyStruct>(...)`)
181    //   and with the GraphQL client's `query::<T>()` which requires `T: DeserializeOwned`
182    let output = quote! {
183        impl #impl_generics #struct_ident #ty_generics #where_clause {
184            pub fn from_value(value: serde_json::Value) -> Result<Self, String> {
185                #(#field_extractions)*
186
187                Ok(Self {
188                    #(#field_names),*
189                })
190            }
191        }
192
193        // TODO: Implement efficient deserialization that only extracts the fields we need.
194        impl<'de> serde::Deserialize<'de> for #struct_ident #ty_generics #where_clause {
195            fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
196            where
197                D: serde::Deserializer<'de>,
198            {
199                let value = serde_json::Value::deserialize(deserializer)?;
200                Self::from_value(value).map_err(serde::de::Error::custom)
201            }
202        }
203    };
204
205    Ok(output)
206}

208/// Generate code to extract a single field from JSON using its path.
209///
210/// Supports multiple path formats:
211/// - Simple: `"object.address"` - navigates to nested field
212/// - Array: `"nodes[].name"` - iterates over array, extracts field from each element
213/// - Nested arrays: `"nodes[].edges[].id"` - nested iteration, returns `Vec<Vec<T>>`
214/// - Aliased: `"alias:field"` - uses alias for JSON extraction, field for validation
215fn generate_field_extraction(
216    path: &path::ParsedPath,
217    type_structure: &validation::TypeStructure,
218    field_ident: &syn::Ident,
219) -> TokenStream2 {
220    let full_path = &path.raw;
221    let inner = generate_from_segments(full_path, &path.segments, type_structure);
222    // The inner expression returns Result<T, String>, so we use ? to unwrap
223    quote! {
224        let #field_ident = {
225            let current = &value;
226            #inner?
227        };
228    }
229}

231/// Recursively generate extraction code by traversing path segments.
232///
233/// For JSON extraction, uses the alias if present, otherwise uses the field name.
234/// Returns code that evaluates to `Result<T, String>` (caller adds `?` to unwrap).
235///
236/// ## Example: `"data.nodes[].edges[].id"` with `Option<Vec<Vec<String>>>`
237///
238/// Each `[]` in the path corresponds to one `Vec<_>` wrapper in the type.
239///
240/// For `Option<_>` types, null at the outer level returns `Ok(None)`. This is achieved
241/// by wrapping the extraction in a closure to capture early returns. However, once
242/// inside an array iteration, the element type (`Vec<String>`) is not Optional, so
243/// null values there return errors instead.
244///
245/// ```ignore
246/// (|| {
247///     // "data" (non-list) - null returns None (outer Optional)
248///     let current = current.get("data").ok_or_else(|| "missing 'data'")?;
249///     if current.is_null() { return Ok(None); }
250///
251///     // "nodes[]" (list) - null returns None (outer Optional)
252///     let field_value = current.get("nodes").ok_or_else(|| "missing 'nodes'")?;
253///     if field_value.is_null() { return Ok(None); }
254///     let array = field_value.as_array().ok_or_else(|| "expected array")?;
255///     array.iter().map(|current| {
256///         // Element type: Vec<String> (not Optional, so null = error)
257///
258///         // "edges[]" (list) - null returns Err
259///         let field_value = current.get("edges").ok_or_else(|| "missing 'edges'")?;
260///         if field_value.is_null() { return Err("null at 'edges'"); }
261///         let array = field_value.as_array().ok_or_else(|| "expected array")?;
262///         array.iter().map(|current| {
263///             // Element type: String (not Optional, so null = error)
264///
265///             // "id" (scalar) - null returns Err
266///             let current = current.get("id").ok_or_else(|| "missing 'id'")?;
267///             if current.is_null() { return Err("null at 'id'"); }
268///             serde_json::from_value(current.clone())
269///         }).collect::<Result<Vec<_>, _>>()
270///     }).collect::<Result<Vec<_>, _>>()
271///     .map(Some)  // Wrap in Some for Option
272/// })()
273/// ```
274fn generate_from_segments(
275    full_path: &str,
276    segments: &[path::PathSegment],
277    type_structure: &validation::TypeStructure,
278) -> TokenStream2 {
279    // Step 1: Check if outer type is Optional and unwrap it
280    let (is_optional, inner_type) = match type_structure {
281        validation::TypeStructure::Optional(inner) => (true, inner.as_ref()),
282        other => (false, other),
283    };
284
285    // Step 2: Generate core extraction code
286    let core = generate_from_segments_core(full_path, segments, inner_type, is_optional);
287
288    // Step 3: Wrap Optional types in a closure to capture `return Ok(None)` from null handling.
289    // Without closure, the return would escape to `from_value`. With closure, field gets `None`.
290    if is_optional {
291        quote! {
292            (|| {
293                #core.map(Some)
294            })()
295        }
296    } else {
297        core
298    }
299}

301/// Core extraction logic that handles both list and non-list segments.
302///
303/// `null_returns_none`: If true, null values cause early return with `Ok(None)`.
304/// This is set when the outermost type is Optional.
305fn generate_from_segments_core(
306    full_path: &str,
307    segments: &[path::PathSegment],
308    type_structure: &validation::TypeStructure,
309    null_returns_none: bool,
310) -> TokenStream2 {
311    // Base case: no more segments, deserialize the current value
312    let Some((segment, rest)) = segments.split_first() else {
313        return quote! {
314            serde_json::from_value(current.clone())
315                .map_err(|e| format!("failed to deserialize '{}': {}", #full_path, e))
316        };
317    };
318
319    let name = segment.field;
320    // Use alias for JSON extraction if present, otherwise use field name
321    let json_key = segment.json_key();
322
323    // Generate null handling based on whether outer type is Optional
324    let on_null = if null_returns_none {
325        quote! { return Ok(None) }
326    } else {
327        quote! {
328            return Err(format!("null value at '{}' in path '{}'", #name, #full_path))
329        }
330    };
331
332    if segment.is_list {
333        // For list segments, unwrap Vector to get element type
334        let element_type = match type_structure {
335            validation::TypeStructure::Vector(inner) => inner.as_ref(),
336            _ => unreachable!("validated: list segment requires Vec type"),
337        };
338
339        // Each array element is processed independently with its own type structure.
340        // Use generate_from_segments (not _core) to handle element-level Optional.
341        let rest_code = generate_from_segments(full_path, rest, element_type);
342
343        quote! {
344            let field_value = current.get(#json_key)
345                .ok_or_else(|| format!("missing field '{}' in path '{}'", #json_key, #full_path))?;
346            if field_value.is_null() {
347                #on_null
348            }
349            let array = field_value.as_array()
350                .ok_or_else(|| format!("expected array at '{}' in path '{}'", #json_key, #full_path))?;
351            array.iter()
352                .map(|current| { #rest_code })
353                .collect::<Result<Vec<_>, String>>()
354        }
355    } else {
356        // For non-list segments, pass type unchanged to handle nested structures
357        let rest_code =
358            generate_from_segments_core(full_path, rest, type_structure, null_returns_none);
359
360        quote! {
361            let current = current.get(#json_key)
362                .ok_or_else(|| format!("missing field '{}' in path '{}'", #json_key, #full_path))?;
363            if current.is_null() {
364                #on_null
365            }
366            #rest_code
367        }
368    }
