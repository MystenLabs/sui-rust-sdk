1//! Procedural macros for sui-graphql with compile-time schema validation.
2//!
3//! This crate provides the `Response` derive macro that:
4//! - Validates field paths against the Sui GraphQL schema at compile time
5//! - Generates deserialization code for extracting nested fields from JSON responses
6//!
7//! # Example
8//!
9//! ```ignore
10//! use sui_graphql_macros::Response;
11//!
12//! #[derive(Response)]
13//! struct ObjectData {
14//!     #[field(path = "object.address")]
15//!     address: String,
16//!     #[field(path = "object.version")]
17//!     version: u64,
18//! }
19//!
20//! // The macro validates paths against the schema and generates extraction code.
21//! // Invalid paths like "object.nonexistent" will cause a compile error.
22//! ```
23

37/// Derive macro for GraphQL response types with nested field extraction.
38///
39/// Use `#[field(path = "...")]` to specify the JSON path to extract each field.
40/// Paths are dot-separated (e.g., `"object.address"` extracts `json["object"]["address"]`).
41///
42/// # Generated Code
43///
44/// The macro generates:
45/// - `from_value(serde_json::Value) -> Result<Self, String>` method
46/// - `Deserialize` implementation that uses `from_value`
47///
48/// # Example
49///
50/// ```ignore
51/// #[derive(Response)]
52/// struct ChainInfo {
53///     #[field(path = "chainIdentifier")]
54///     chain_id: String,
55///
56///     #[field(path = "epoch.epochId")]
57///     epoch_id: Option<u64>,
58/// }
59/// ```
60#[proc_macro_derive(Response, attributes(field))]
61pub fn derive_query_response(input: TokenStream) -> TokenStream {
62    let input = parse_macro_input!(input as DeriveInput);
63
64    match derive_query_response_impl(input) {
65        Ok(tokens) => tokens.into(),
66        Err(err) => err.to_compile_error().into(),
67    }
68}
69
70fn derive_query_response_impl(input: DeriveInput) -> Result<TokenStream2, syn::Error> {
71    // Darling input structures - define the "schema" for macro input.
72    // Darling generates parsing code automatically, including error messages for invalid input.
73
74    #[derive(Debug, FromDeriveInput)]
75    #[darling(supports(struct_named))] // Only supports structs with named fields
76    struct ResponseInput {
77        ident: syn::Ident,                           // Struct name
78        generics: syn::Generics,                     // Generic parameters
79        data: darling::ast::Data<(), ResponseField>, // Struct fields
80    }
81
82    #[derive(Debug, FromField)]
83    #[darling(attributes(field))] // Parse #[field(...)] attributes
84    struct ResponseField {
85        ident: Option<syn::Ident>, // Field name
86        path: String,              // The path = "..." value
87        #[darling(default)]
88        skip_validation: bool, // Skip schema validation if true
89    }
90
91    let parsed = ResponseInput::from_derive_input(&input)?;
92
93    // Load the GraphQL schema for validation
94    let schema = schema::Schema::load()?;
95
96    let fields = parsed
97        .data
98        .as_ref()
99        .take_struct()
100        .ok_or_else(|| syn::Error::new_spanned(&input, "Response only supports structs"))?
101        .fields;
102
103    // Generate extraction code for each field
104    let mut field_extractions = Vec::new();
105    let mut field_names = Vec::new();
106
107    for field in &fields {
108        let field_ident = field
109            .ident
110            .as_ref()
111            .ok_or_else(|| syn::Error::new_spanned(&input, "Unnamed fields not supported"))?;
112
113        let path = &field.path;
114
115        // Validate the path against the GraphQL schema (unless skip_validation is set)
116        if !field.skip_validation {
117            validation::validate_path(schema, path, field_ident)?;
118        }
119
120        let extraction = generate_field_extraction(path, field_ident);
121        field_extractions.push(extraction);
122        field_names.push(field_ident);
123    }
124
125    let struct_ident = &parsed.ident;
126    let (impl_generics, ty_generics, where_clause) = parsed.generics.split_for_impl();
127
128    // Generate both `from_value` and `Deserialize` impl:
129    //
130    // - `from_value`: Core extraction logic, parses from serde_json::Value
131    // - `Deserialize`: Allows direct use with serde (e.g., `serde_json::from_str::<MyStruct>(...)`)
132    //   and with the GraphQL client's `query::<T>()` which requires `T: DeserializeOwned`
133    let output = quote! {
134        impl #impl_generics #struct_ident #ty_generics #where_clause {
135            pub fn from_value(value: serde_json::Value) -> Result<Self, String> {
136                #(#field_extractions)*
137
138                Ok(Self {
139                    #(#field_names),*
140                })
141            }
142        }
143
144        // TODO: Implement efficient deserialization that only extracts the fields we need.
145        impl<'de> serde::Deserialize<'de> for #struct_ident #ty_generics #where_clause {
146            fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
147            where
148                D: serde::Deserializer<'de>,
149            {
150                let value = serde_json::Value::deserialize(deserializer)?;
151                Self::from_value(value).map_err(serde::de::Error::custom)
152            }
153        }
154    };
155
156    Ok(output)
157}

159/// A segment in a field path.
160///
161/// Paths like `"data.nodes[].name"` are parsed into segments:
162/// - `PathSegment { field: "data", is_array: false }`
163/// - `PathSegment { field: "nodes", is_array: true }`
164/// - `PathSegment { field: "name", is_array: false }`
165struct PathSegment<'a> {
166    /// The field name to access
167    field: &'a str,
168    /// Whether this is an array field (ends with `[]`)
169    is_array: bool,
170}
171
172/// Parse a path string into segments.
173///
174/// Each dot-separated part becomes a `PathSegment`. If it ends with `[]`, it's an array field.
175fn parse_path(path: &str) -> Vec<PathSegment<'_>> {
176    path.split('.')
177        .map(|segment| {
178            let (field, is_array) = if let Some(stripped) = segment.strip_suffix("[]") {
179                (stripped, true)
180            } else {
181                (segment, false)
182            };
183            PathSegment { field, is_array }
184        })
185        .collect()
186}

188/// Generate code to extract a single field from JSON using its path.
189///
190/// Supports multiple path formats:
191/// - Simple: `"object.address"` - navigates to nested field
192/// - Array: `"nodes[].name"` - iterates over array, extracts field from each element
193/// - Nested arrays: `"nodes[].edges[].id"` - nested iteration, returns `Vec<Vec<T>>`
194fn generate_field_extraction(path: &str, field_ident: &syn::Ident) -> TokenStream2 {
195    let segments = parse_path(path);
196    let inner = generate_from_segments(path, &segments);
197    // The inner expression returns Result<T, String>, so we use ? to unwrap
198    quote! {
199        let #field_ident = {
200            let current = &value;
201            #inner?
202        };
203    }
204}
205
206/// Recursively generate extraction code by traversing path segments.
207///
208/// Returns code that evaluates to `Result<T, String>` (caller adds `?` to unwrap).
209///
210/// ## Example: Simple path `"object.address"`
211///
212/// ```ignore
213/// let current = current.get("object").ok_or_else(|| ...)?;
214/// if current.is_null() { return serde_json::from_value(null)... }
215/// let current = current.get("address").ok_or_else(|| ...)?;
216/// serde_json::from_value(current.clone()).map_err(|e| ...)
217/// ```
218///
219/// ## Example: Array path `"nodes[].name"` (returns `Option<Vec<T>>`)
220///
221/// ```ignore
222/// {
223///     let field_value = current.get("nodes").ok_or_else(|| ...)?;
224///     if field_value.is_null() {
225///         Ok(None)
226///     } else {
227///         let array = field_value.as_array().ok_or_else(|| ...)?;
228///         let vec: Vec<_> = array.iter().map(|current| { ... }).collect::<Result<_, _>>()?;
229///         Ok(Some(vec))
230///     }
231/// }
232/// ```
233///
234/// TODO: Add support for `?` syntax in paths (e.g., "object?.field") to handle nullable
235/// fields efficiently without requiring Option wrappers at every level.
236fn generate_from_segments(full_path: &str, segments: &[PathSegment<'_>]) -> TokenStream2 {
237    // Base case: no more segments, deserialize the current value
238    if segments.is_empty() {
239        return quote! {
240            serde_json::from_value(current.clone())
241                .map_err(|e| format!("failed to deserialize '{}': {}", #full_path, e))
242        };
243    }
244
245    let segment = &segments[0];
246    let name = segment.field;
247    let rest = generate_from_segments(full_path, &segments[1..]);
248
249    if segment.is_array {
250        quote! {
251            {
252                let field_value = current.get(#name)
253                    .ok_or_else(|| format!("missing field '{}' in path '{}'", #name, #full_path))?;
254                if field_value.is_null() {
255                    Ok(None)
256                } else {
257                    let array = field_value.as_array()
258                        .ok_or_else(|| format!("expected array at '{}' in path '{}'", #name, #full_path))?;
259                    array.iter()
260                        .map(|current| { #rest })
261                        .collect::<Result<Vec<_>, String>>()
262                        .map(Some)
263                }
264            }
265        }
266    } else {
267        quote! {
268            {
269                let current = current.get(#name)
270                    .ok_or_else(|| format!("missing field '{}' in path '{}'", #name, #full_path))?;
271                // If null, skip remaining navigation and let serde handle it
272                // (returns Ok(None) for Option<T>, error for non-Option)
273                if current.is_null() {
274                    serde_json::from_value(current.clone())
275                        .map_err(|e| format!("failed to deserialize '{}': {}", #full_path, e))
276                } else {
277                    #rest
278                }
279            }
280        }
281    }
