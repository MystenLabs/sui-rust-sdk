1//! Procedural macros for sui-graphql with compile-time schema validation.
2//!
3//! This crate provides the `Response` derive macro that:
4//! - Validates field paths against the Sui GraphQL schema at compile time
5//! - Generates deserialization code for extracting nested fields from JSON responses
6//!
7//! # Example
8//!
9//! ```ignore
10//! use sui_graphql_macros::Response;
11//!
12//! #[derive(Response)]
13//! struct ObjectData {
14//!     #[field(path = "object.address")]
15//!     address: String,
16//!     #[field(path = "object.version")]
17//!     version: u64,
18//! }
19//!
20//! // The macro validates paths against the schema and generates extraction code.
21//! // Invalid paths like "object.nonexistent" will cause a compile error.
22//! ```

45#[derive(Debug, FromDeriveInput)]
46#[darling(attributes(response), supports(struct_named, enum_newtype))]
47struct ResponseInput {
48    ident: syn::Ident,
49    generics: syn::Generics,
50    data: darling::ast::Data<ResponseVariant, ResponseField>,
51    #[darling(default)]
52    schema: Option<String>,
53    #[darling(default)]
54    root_type: Option<SpannedValue<String>>,
55}

57/// A struct field (requires `#[field(path = "...")]`).
58#[derive(Debug, FromField)]
59#[darling(attributes(field))]
60struct ResponseField {
61    ident: Option<syn::Ident>,
62    ty: syn::Type,
63    path: SpannedValue<String>,
64    #[darling(default)]
65    skip_schema_validation: bool,
66}

68/// The inner type of a newtype enum variant.
69#[derive(Debug, FromField)]
70struct VariantInner {
71    ty: syn::Type,

74/// An enum variant mapping to a GraphQL union member.
75#[derive(Debug, FromVariant)]
76#[darling(attributes(response))]
77struct ResponseVariant {
78    ident: syn::Ident,
79    fields: darling::ast::Fields<VariantInner>,
80    /// The GraphQL type name this variant maps to (e.g., `#[response(on = "MoveValue")]`).
81    /// Defaults to the variant ident if not specified.
82    #[darling(default)]
83    on: Option<SpannedValue<String>>,
84}

86/// Derive macro for GraphQL response types with nested field extraction.
87///
88/// Use `#[field(path = "...")]` to specify the JSON path to extract each field.
89/// Paths are dot-separated (e.g., `"object.address"` extracts `json["object"]["address"]`).
90///
91/// # Root Type
92///
93/// By default, field paths are validated against the `Query` type. Use
94/// `#[response(root_type = "...")]` to validate against a different type instead.
95///
96/// # Generated Code
97///
98/// The macro generates:
99/// - `from_value(serde_json::Value) -> Result<Self, String>` method
100/// - `Deserialize` implementation that uses `from_value`
101///
102/// # Example
103///
104/// ```ignore
105/// // Query response (default)
106/// #[derive(Response)]
107/// struct ChainInfo {
108///     #[field(path = "chainIdentifier")]
109///     chain_id: String,
110///
111///     #[field(path = "epoch.epochId")]
112///     epoch_id: Option<u64>,
113/// }
114///
115/// // Mutation response
116/// #[derive(Response)]
117/// #[response(root_type = "Mutation")]
118/// struct ExecuteResult {
119///     #[field(path = "executeTransaction.effects.effectsBcs")]
120///     effects_bcs: Option<String>,
121/// }
122/// ```
123#[proc_macro_derive(Response, attributes(response, field))]
124pub fn derive_query_response(input: TokenStream) -> TokenStream {
125    let input = parse_macro_input!(input as DeriveInput);
126
127    match derive_query_response_impl(input) {
128        Ok(tokens) => tokens.into(),
129        Err(err) => err.to_compile_error().into(),
130    }
131}
132
133fn derive_query_response_impl(input: DeriveInput) -> Result<TokenStream2, syn::Error> {
134    let parsed = ResponseInput::from_derive_input(&input)?;
135
136    // Load the GraphQL schema for validation.
137    // If a custom schema path is provided, load it; otherwise use the embedded Sui schema.
138    let loaded_schema = if let Some(path) = &parsed.schema {
139        // Resolve path relative to the crate's directory.
140        // SUI_GRAPHQL_SCHEMA_DIR is used by trybuild tests (which run from a temp directory).
141        let base_dir = std::env::var("SUI_GRAPHQL_SCHEMA_DIR")
142            .or_else(|_| std::env::var("CARGO_MANIFEST_DIR"))
143            .unwrap();
144        let full_path = std::path::Path::new(&base_dir).join(path);
145        let sdl = std::fs::read_to_string(&full_path).map_err(|e| {
146            syn::Error::new(
147                proc_macro2::Span::call_site(),
148                format!(
149                    "Failed to read schema from '{}': {}",
150                    full_path.display(),
151                    e
152                ),
153            )
154        })?;
155        Some(schema::Schema::from_sdl(&sdl)?)
156    } else {
157        None
158    };
159    let schema = if let Some(schema) = &loaded_schema {
160        schema
161    } else {
162        schema::Schema::load()?
163    };
164
165    // Determine root type: use specified root_type or default to "Query"
166    let root_type = parsed
167        .root_type
168        .as_ref()
169        .map(|s| s.as_str())
170        .unwrap_or("Query");
171
172    // Validate that the root type exists in the schema
173    if !schema.has_type(root_type) {
174        use std::fmt::Write;
175
176        let type_names = schema.type_names();
177        let suggestion = validation::find_similar(&type_names, root_type);
178
179        let mut msg = format!("Type '{}' not found in GraphQL schema", root_type);
180        if let Some(suggested) = suggestion {
181            write!(msg, ". Did you mean '{}'?", suggested).unwrap();
182        }
183
184        // We only enter this block if root_type was explicitly specified (and invalid),
185        // since "Query" (the default) always exists in a valid schema.
186        let span = parsed.root_type.as_ref().unwrap().span();
187
188        return Err(syn::Error::new(span, msg));
189    }
190
191    match parsed.data {
192        darling::ast::Data::Struct(ref fields) => {
193            generate_struct_impl(&parsed, &fields.fields, schema, root_type)
194        }
195        darling::ast::Data::Enum(ref variants) => {
196            generate_enum_impl(&parsed, variants, schema, root_type)
197        }
198    }
199}

201/// Generate `from_value` and `Deserialize` for a struct.
202fn generate_struct_impl(
203    input: &ResponseInput,
204    fields: &[ResponseField],
205    schema: &schema::Schema,
206    root_type: &str,
207) -> Result<TokenStream2, syn::Error> {
208    let ident = &input.ident;
209    let (impl_generics, ty_generics, where_clause) = input.generics.split_for_impl();
210
211    // Generate extraction code for each field
212    let mut field_extractions = Vec::new();
213    let mut field_names = Vec::new();
214
215    for field in fields {
216        let field_ident = field
217            .ident
218            .as_ref()
219            .expect("darling ensures named fields only");
220
221        let spanned_path = &field.path;
222        let parsed_path = path::ParsedPath::parse(spanned_path.as_str())
223            .map_err(|e| syn::Error::new(spanned_path.span(), e.to_string()))?;
224
225        let terminal_type = if !field.skip_schema_validation {
226            Some(validation::validate_path_against_schema(
227                schema,
228                root_type,
229                &parsed_path,
230                spanned_path.span(),
231            )?)
232        } else {
233            None
234        };
235
236        // Skip Vec excess check when schema validation is skipped (user takes full
237        // responsibility) or when the terminal type is an object-like scalar (e.g., JSON)
238        // whose value can be an array.
239        let skip_vec_excess_check = field.skip_schema_validation
240            || terminal_type.is_some_and(validation::is_object_like_scalar);
241        validation::validate_type_matches_path(&parsed_path, &field.ty, skip_vec_excess_check)?;
242
243        // Generate extraction code using the same parsed path
244        let type_structure = validation::analyze_type(&field.ty);
245        let extraction = generate_field_extraction(&parsed_path, &type_structure, field_ident);
246        field_extractions.push(extraction);
247        field_names.push(field_ident);
248    }
249
250    // Generate both `from_value` and `Deserialize` impl:
251    //
252    // - `from_value`: Core extraction logic, parses from serde_json::Value
253    // - `Deserialize`: Allows direct use with serde (e.g., `serde_json::from_str::<MyStruct>(...)`)
254    //   and with the GraphQL client's `query::<T>()` which requires `T: DeserializeOwned`
255    Ok(quote! {
256        impl #impl_generics #ident #ty_generics #where_clause {
257            pub fn from_value(value: serde_json::Value) -> Result<Self, String> {
258                #(#field_extractions)*
259
260                Ok(Self {
261                    #(#field_names),*
262                })
263            }
264        }
265
266        // TODO: Implement efficient deserialization that only extracts the fields we need.
267        impl<'de> serde::Deserialize<'de> for #ident #ty_generics #where_clause {
268            fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
269            where
270                D: serde::Deserializer<'de>,
271            {
272                let value = serde_json::Value::deserialize(deserializer)?;
273                Self::from_value(value).map_err(serde::de::Error::custom)
274            }
275        }
276    })
277}

279/// Generate `from_value` and `Deserialize` for an enum (GraphQL union).
280///
281/// Each variant wraps a type that implements `from_value`. Dispatches on `__typename`.
282fn generate_enum_impl(
283    input: &ResponseInput,
284    variants: &[ResponseVariant],
285    schema: &schema::Schema,
286    root_type: &str,
287) -> Result<TokenStream2, syn::Error> {
288    let ident = &input.ident;
289    let (impl_generics, ty_generics, where_clause) = input.generics.split_for_impl();
290
291    let root_type_span = input
292        .root_type
293        .as_ref()
294        .map(|s| s.span())
295        .unwrap_or_else(|| ident.span());
296
297    if !schema.is_union(root_type) {
298        return Err(syn::Error::new(
299            root_type_span,
300            format!(
301                "'{}' is not a union type. \
302                 Enum Response requires root_type to be a GraphQL union",
303                root_type
304            ),
305        ));
306    }
307
308    let mut match_arms = Vec::new();
309
310    for variant in variants {
311        let variant_ident = &variant.ident;
312
313        // Resolve the GraphQL typename: explicit `on` or variant ident
314        let graphql_typename = variant
315            .on
316            .as_ref()
317            .map(|s| s.as_str().to_string())
318            .unwrap_or_else(|| variant_ident.to_string());
319
320        let span = variant
321            .on
322            .as_ref()
323            .map(|s| s.span())
324            .unwrap_or_else(|| variant_ident.span());
325
326        if let Err(mut err) =
327            validation::validate_union_member(schema, root_type, &graphql_typename, span)
328        {
329            if variant.on.is_none() {
330                err.combine(syn::Error::new(
331                    span,
332                    "hint: use #[response(on = \"...\")] to specify a GraphQL type name different from the variant name",
333                ));
334            }
335            return Err(err);
336        }
337
338        // Newtype variant: delegate to inner type's from_value
339        let inner_ty = &variant.fields.fields[0].ty;
340        match_arms.push(quote! {
341            #graphql_typename => {
342                Ok(Self::#variant_ident(
343                    <#inner_ty>::from_value(value)?
344                ))
345            }
346        });
347    }
348
349    let root_type_str = root_type;
350    let enum_name_str = ident.to_string();
351
352    Ok(quote! {
353        impl #impl_generics #ident #ty_generics #where_clause {
354            pub fn from_value(value: serde_json::Value) -> Result<Self, String> {
355                let typename = value.get("__typename")
356                    .and_then(|v| v.as_str())
357                    .ok_or_else(|| format!(
358                        "union '{}' requires '__typename' in the response to distinguish variants. \
359                         Make sure your query requests '__typename' on this field ({})",
360                        #root_type_str, #enum_name_str
361                    ))?;
362
363                match typename {
364                    #(#match_arms)*
365                    other => Err(format!(
366                        "unknown __typename '{}' for union '{}' ({})",
367                        other, #root_type_str, #enum_name_str
368                    )),
369                }
370            }
371        }
372
373        impl<'de> serde::Deserialize<'de> for #ident #ty_generics #where_clause {
374            fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
375            where
376                D: serde::Deserializer<'de>,
377            {
378                let value = serde_json::Value::deserialize(deserializer)?;
379                Self::from_value(value).map_err(serde::de::Error::custom)
380            }
381        }
382    })
383}

385/// Generate code to extract a single field from JSON using its path.
386///
387/// Supports multiple path formats:
388/// - Simple: `"object.address"` - navigates to nested field
389/// - Array: `"nodes[].name"` - iterates over array, extracts field from each element
390/// - Nested arrays: `"nodes[].edges[].id"` - nested iteration, returns `Vec<Vec<T>>`
391/// - Aliased: `"alias:field"` - uses alias for JSON extraction, field for validation
392fn generate_field_extraction(
393    path: &path::ParsedPath,
394    type_structure: &validation::TypeStructure,
395    field_ident: &syn::Ident,
396) -> TokenStream2 {
397    let full_path = &path.raw;
398    let inner = generate_from_segments(full_path, &path.segments, type_structure);
399    // The inner expression returns Result<T, String>, so we use ? to unwrap
400    quote! {
401        let #field_ident = {
402            let current = &value;
403            #inner?
404        };
405    }
406}
407
408/// Recursively generate extraction code by traversing path segments.
409///
410/// For JSON extraction, uses the alias if present, otherwise uses the field name.
411/// Returns code that evaluates to `Result<T, String>` (caller adds `?` to unwrap).
412///
413/// ## Example: `"data.nodes[].edges[].id"` with `Option<Vec<Vec<String>>>`
414///
415/// Each `[]` in the path corresponds to one `Vec<_>` wrapper in the type.
416///
417/// For `Option<_>` types, null at the outer level returns `Ok(None)`. This is achieved
418/// by wrapping the extraction in a closure to capture early returns. However, once
419/// inside an array iteration, the element type (`Vec<String>`) is not Optional, so
420/// null values there return errors instead.
421///
422/// ```ignore
423/// (|| {
424///     // "data" (non-list) - missing/null returns None (outer Optional)
425///     let current = current.get("data").unwrap_or(&serde_json::Value::Null);
426///     if current.is_null() { return Ok(None); }
427///
428///     // "nodes[]" (list) - missing/null returns None (outer Optional)
429///     let field_value = current.get("nodes").unwrap_or(&serde_json::Value::Null);
430///     if field_value.is_null() { return Ok(None); }
431///     let array = field_value.as_array().ok_or_else(|| "expected array")?;
432///     array.iter().map(|current| {
433///         // Element type: Vec<String> (not Optional, so null = error)
434///
435///         // "edges[]" (list) - missing/null returns Err
436///         let field_value = current.get("edges").unwrap_or(&serde_json::Value::Null);
437///         if field_value.is_null() { return Err("null at 'edges'"); }
438///         let array = field_value.as_array().ok_or_else(|| "expected array")?;
439///         array.iter().map(|current| {
440///             // Element type: String (not Optional, so null = error)
441///
442///             // "id" (scalar) - missing/null returns Err
443///             let current = current.get("id").unwrap_or(&serde_json::Value::Null);
444///             if current.is_null() { return Err("null at 'id'"); }
445///             serde_json::from_value(current.clone())
446///         }).collect::<Result<Vec<_>, _>>()
447///     }).collect::<Result<Vec<_>, _>>()
448///     .map(Some)  // Wrap in Some for Option
449/// })()
450/// ```
451fn generate_from_segments(
452    full_path: &str,
453    segments: &[path::PathSegment],
454    type_structure: &validation::TypeStructure,
455) -> TokenStream2 {
456    // Step 1: Check if outer type is Optional and unwrap it
457    let (is_optional, inner_type) = match type_structure {
458        validation::TypeStructure::Optional(inner) => (true, inner.as_ref()),
459        other => (false, other),
460    };
461
462    // Step 2: Generate core extraction code
463    let core = generate_from_segments_core(full_path, segments, inner_type, is_optional);
464
465    // Step 3: Wrap Optional types in a closure to capture `return Ok(None)` from null handling.
466    // Without closure, the return would escape to `from_value`. With closure, field gets `None`.
467    if is_optional {
468        quote! {
469            (|| {
470                #core.map(Some)
471            })()
472        }
473    } else {
474        core
475    }
476}

478/// Core extraction logic that handles both list and non-list segments.
479///
480/// `null_returns_none`: If true, null values cause early return with `Ok(None)`.
481/// This is set when the outermost type is Optional.
482fn generate_from_segments_core(
483    full_path: &str,
484    segments: &[path::PathSegment],
485    type_structure: &validation::TypeStructure,
486    null_returns_none: bool,
487) -> TokenStream2 {
488    // Base case: no more segments, deserialize the current value
489    let Some((segment, rest)) = segments.split_first() else {
490        return quote! {
491            serde_json::from_value(current.clone())
492                .map_err(|e| format!("failed to deserialize '{}': {}", #full_path, e))
493        };
494    };
495
496    let name = segment.field;
497    // Use alias for JSON extraction if present, otherwise use field name
498    let json_key = segment.json_key();
499
500    // Generate null handling based on whether outer type is Optional
501    let on_null = if null_returns_none {
502        quote! { return Ok(None) }
503    } else {
504        quote! {
505            return Err(format!("null value at '{}' in path '{}'", #name, #full_path))
506        }
507    };
508
509    if segment.is_list {
510        // For list segments, unwrap Vector to get element type
511        let element_type = match type_structure {
512            validation::TypeStructure::Vector(inner) => inner.as_ref(),
513            _ => unreachable!("validated: list segment requires Vec type"),
514        };
515
516        // Each array element is processed independently with its own type structure.
517        // Use generate_from_segments (not _core) to handle element-level Optional.
518        let rest_code = generate_from_segments(full_path, rest, element_type);
519
520        quote! {
521            // Treat missing fields as null (allows Option<T> to deserialize as None)
522            let field_value = current.get(#json_key).unwrap_or(&serde_json::Value::Null);
523            if field_value.is_null() {
524                #on_null
525            }
526            let array = field_value.as_array()
527                .ok_or_else(|| format!("expected array at '{}' in path '{}'", #json_key, #full_path))?;
528            array.iter()
529                .map(|current| { #rest_code })
530                .collect::<Result<Vec<_>, String>>()
531        }
532    } else {
533        // For non-list segments, pass type unchanged to handle nested structures
534        let rest_code =
535            generate_from_segments_core(full_path, rest, type_structure, null_returns_none);
536
537        quote! {
538            // Treat missing fields as null (allows Option<T> to deserialize as None)
539            let current = current.get(#json_key).unwrap_or(&serde_json::Value::Null);
540            if current.is_null() {
541                #on_null
542            }
543            #rest_code
544        }
545    }
