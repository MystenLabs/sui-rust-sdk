1//! Field path validation against the GraphQL schema and Rust types.
2//!
3//! Validates paths and determines iteration by matching:
4//! - Schema: which fields are lists
5//! - Type: how many Vec wrappers
6//!
7//! Iteration happens automatically when schema says list AND type has Vec.
8
9use crate::path::ParsedPath;
10use crate::schema::Schema;

12/// Represents the nesting structure of `Option` and `Vec` in a field type.
13#[derive(Debug, Clone, PartialEq)]
14pub enum TypeStructure {
15    /// A type that is neither `Option` nor `Vec` (e.g., `String`, `u64`)
16    Plain,
17    /// `Option<T>` wrapping an inner structure
18    Optional(Box<TypeStructure>),
19    /// `Vec<T>` wrapping an inner structure
20    Vector(Box<TypeStructure>),
21}
22
23/// Analyze a `syn::Type` into a `TypeStructure`.
24///
25/// Note: Type detection uses simple name matching (e.g., `ident == "Option"`),
26/// the same approach used by serde_derive. This works for standard library types
27/// but won't distinguish custom types with the same name.
28/// See: https://github.com/serde-rs/serde/blob/master/serde_derive/src/internals/attr.rs
29pub fn analyze_type(ty: &syn::Type) -> TypeStructure {
30    if let Some(inner) = unwrap_option(ty) {
31        TypeStructure::Optional(Box::new(analyze_type(inner)))
32    } else if let Some(inner) = unwrap_vec(ty) {
33        TypeStructure::Vector(Box::new(analyze_type(inner)))
34    } else {
35        TypeStructure::Plain
36    }
37}
38
39/// Returns the inner type if `ty` is `Option<T>`, otherwise `None`.
40fn unwrap_option(ty: &syn::Type) -> Option<&syn::Type> {
41    unwrap_type(ty, "Option")
42}
43
44/// Returns the inner type if `ty` is `Vec<T>`, otherwise `None`.
45fn unwrap_vec(ty: &syn::Type) -> Option<&syn::Type> {
46    unwrap_type(ty, "Vec")
47}

49/// Returns the inner type if `ty` matches `TypeName<T>`, otherwise `None`.
50fn unwrap_type<'a>(ty: &'a syn::Type, type_name: &str) -> Option<&'a syn::Type> {
51    let syn::Type::Path(type_path) = ungroup(ty) else {
52        return None;
53    };
54    let seg = type_path.path.segments.last()?;
55    let syn::PathArguments::AngleBracketed(args) = &seg.arguments else {
56        return None;
57    };
58
59    if seg.ident == type_name
60        && args.args.len() == 1
61        && let syn::GenericArgument::Type(inner) = &args.args[0]
62    {
63        return Some(inner);
64    }
65    None
66}

68/// Unwrap `syn::Type::Group` nodes which may appear in macro-generated code.
69///
70/// When a macro captures a type with `$t:ty` and substitutes it, the type may be
71/// wrapped in an invisible `Group` node. For example, `Option<String>` might appear
72/// as `Group(Path("Option<String>"))` instead of just `Path("Option<String>")`.
73///
74/// Credit: serde_derive (https://github.com/serde-rs/serde)
75fn ungroup(mut ty: &syn::Type) -> &syn::Type {
76    while let syn::Type::Group(group) = ty {
77        ty = &group.elem;
78    }
79    ty
80}

82/// Count the number of `Vec` wrappers in a type structure.
83pub fn count_vec_depth(ts: &TypeStructure) -> usize {
84    match ts {
85        TypeStructure::Plain => 0,
86        TypeStructure::Optional(inner) => count_vec_depth(inner),
87        TypeStructure::Vector(inner) => 1 + count_vec_depth(inner),
88    }
89}

91/// Scalars whose values can be objects or arrays in JSON.
92///
93/// These scalars represent structured data (not simple strings or numbers), so the macro
94/// cannot validate Vec count â€” the Rust type is the user's responsibility.
95const OBJECT_LIKE_SCALARS: &[&str] = &[
96    "JSON",
97    "MoveTypeLayout",
98    "MoveTypeSignature",
99    "OpenMoveTypeSignature",
100];

102/// Validate a parsed field path against the schema.
103///
104/// Checks that all fields in the path exist in the schema and that `[]` markers
105/// match the schema's list types. List fields must always use `[]` explicitly.
106///
107/// Returns the terminal type name (the type of the last field in the path).
108pub fn validate_path_against_schema<'a>(
109    schema: &'a Schema,
110    root_type: &'a str,
111    path: &ParsedPath,
112    span: proc_macro2::Span,
113) -> Result<&'a str, syn::Error> {
114    let mut current_type: &str = root_type;
115
116    for segment in &path.segments {
117        let field = schema
118            .get_field(current_type, segment.field)
119            .ok_or_else(|| field_not_found_error(schema, current_type, segment.field, span))?;
120
121        if segment.is_list && !field.is_list {
122            return Err(syn::Error::new(
123                span,
124                format!(
125                    "Cannot use '[]' on non-list field '{}' (type '{}')",
126                    segment.field, field.type_name
127                ),
128            ));
129        }
130
131        if !segment.is_list && field.is_list {
132            return Err(syn::Error::new(
133                span,
134                format!(
135                    "Field '{}' is a list type, use '{}[]' to iterate over it",
136                    segment.field, segment.field
137                ),
138            ));
139        }
140
141        current_type = &field.type_name;
142    }
143
144    Ok(current_type)
145}

147/// Validate that a type name is a member of a union.
148pub fn validate_union_member(
149    schema: &Schema,
150    union_type: &str,
151    member_name: &str,
152    span: proc_macro2::Span,
153) -> Result<(), syn::Error> {
154    let union_types = schema.union_types(union_type);
155    if !union_types.contains(&member_name) {
156        let suggestion = find_similar(&union_types, member_name);
157
158        let mut msg = format!(
159            "'{}' is not a member of union '{}'. Members: {}",
160            member_name,
161            union_type,
162            union_types.join(", ")
163        );
164        if let Some(s) = suggestion {
165            msg.push_str(&format!(". Did you mean '{}'?", s));
166        }
167        return Err(syn::Error::new(span, msg));
168    }
169    Ok(())
170}

177/// Validate that the type's Vec count matches the number of list fields in the path.
178///
179/// When `skip_vec_excess_check` is true, extra Vec wrappers beyond the list count are
180/// allowed. This is used for object-like scalars (e.g., JSON) whose values can be arrays.
181///
182/// # Errors
183///
184/// - If Vec count < list count: points to the specific list field missing a Vec
185/// - If Vec count > list count (and `skip_vec_excess_check` is false): too many Vec wrappers
186pub fn validate_type_matches_path(
187    path: &ParsedPath<'_>,
188    ty: &syn::Type,
189    skip_vec_excess_check: bool,
190) -> Result<(), syn::Error> {
191    let type_structure = analyze_type(ty);
192    let vec_count = count_vec_depth(&type_structure);
193    let list_count = path.list_fields().len();
194
195    if vec_count < list_count {
196        let list_fields = path.list_fields();
197        let mismatched_field = list_fields[vec_count];
198        return Err(syn::Error::new_spanned(
199            ty,
200            format!(
201                "field '{}' is a list but type has no Vec wrapper for it",
202                mismatched_field
203            ),
204        ));
205    }
206
207    if !skip_vec_excess_check && vec_count > list_count {
208        return Err(syn::Error::new_spanned(
209            ty,
210            format!(
211                "type has {} Vec wrapper(s) but path '{}' has {} list field(s)",
212                vec_count, path.raw, list_count
213            ),
214        ));
215    }
216
217    Ok(())
218}

220/// Generate an error for a field not found, with "Did you mean?" suggestion.
221fn field_not_found_error(
222    schema: &Schema,
223    type_name: &str,
224    field_name: &str,
225    span: proc_macro2::Span,
226) -> syn::Error {
227    let available = schema.field_names(type_name);
228    let suggestion = find_similar(&available, field_name);
229
230    let mut msg = format!("Field '{field_name}' not found on type '{type_name}'");
231
232    if let Some(suggested) = suggestion {
233        msg.push_str(&format!(". Did you mean '{suggested}'?"));
234    } else if !available.is_empty() {
235        // Sort for deterministic error messages (HashMap iteration order is random)
236        let mut fields: Vec<_> = available;
237        fields.sort();
238        let fields_str = fields.join(", ");
239        msg.push_str(&format!(". Available fields: {fields_str}"));
240    }
241
242    syn::Error::new(span, msg)
243}

245/// Find a similar string using Levenshtein distance.
246pub fn find_similar<'a>(candidates: &[&'a str], target: &str) -> Option<&'a str> {
247    candidates
248        .iter()
249        .filter_map(|&candidate| {
250            let distance = levenshtein_distance(candidate, target);
251            // Only suggest if distance is reasonable (less than half the target length + 1)
252            if distance <= target.len() / 2 + 1 {
253                Some((candidate, distance))
254            } else {
255                None
256            }
257        })
258        .min_by_key(|(_, d)| *d)
259        .map(|(c, _)| c)
260}
261
262/// Simple Levenshtein distance implementation.
263fn levenshtein_distance(a: &str, b: &str) -> usize {
264    let a_chars: Vec<char> = a.chars().collect();
265    let b_chars: Vec<char> = b.chars().collect();
266
267    let m = a_chars.len();
268    let n = b_chars.len();
269
270    if m == 0 {
271        return n;
272    }
273    if n == 0 {
274        return m;
275    }
276
277    let mut dp = vec![vec![0; n + 1]; m + 1];
278
279    #[allow(clippy::needless_range_loop)]
280    for i in 0..=m {
281        dp[i][0] = i;
282    }
283    #[allow(clippy::needless_range_loop)]
284    for j in 0..=n {
285        dp[0][j] = j;
286    }
287
288    for i in 1..=m {
289        for j in 1..=n {
290            let cost = if a_chars[i - 1] == b_chars[j - 1] {
291                0
292            } else {
293                1
294            };
295            dp[i][j] = (dp[i - 1][j] + 1)
296                .min(dp[i][j - 1] + 1)
297                .min(dp[i - 1][j - 1] + cost);
298        }
299    }
300
301    dp[m][n]
302}
