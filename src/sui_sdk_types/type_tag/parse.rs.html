21// static ALLOWED_IDENTIFIERS: &str = r"(?:[a-zA-Z][a-zA-Z0-9_]*)|(?:_[a-zA-Z0-9_]+)";
22static MAX_IDENTIFIER_LENGTH: usize = 128;
23
24/// Maximum nesting depth for type tags. Each `vector<...>` or struct generic
25/// type parameter counts as one level. The bound is inclusive of depth 0:
26/// a type tag at depth 0 (the outermost) is one of `MAX_TYPE_TAG_DEPTH`
27/// allowed levels, so `depth >= MAX_TYPE_TAG_DEPTH` is rejected. Shared
28/// with the BCS deserialization path in `super::serialization` so both
29/// formats reject the same set of pathological inputs.
30pub(super) static MAX_TYPE_TAG_DEPTH: usize = 128;
31
32/// Maximum number of `TypeTag` nodes in a single parsed/deserialized type.
33/// Each instance of `TypeTag` (primitive, vector, or struct) counts as one
34/// node. Catches "wide" types — e.g. a struct with hundreds of generic
35/// parameters at low depth — that the depth bound alone misses. Shared
36/// with the BCS deserialization path so both formats reject the same
37/// set of pathological inputs.
38pub(super) static MAX_TYPE_NODE_COUNT: u64 = 256;

40pub(super) fn parse_identifier(mut input: &str) -> ModalResult<&str> {
41    (identifier, eof).take().parse_next(&mut input)
42}
43
44fn identifier<'s>(input: &mut &'s str) -> ModalResult<&'s str> {
45    alt((
46        (one_of(|c: char| c.is_alpha()), valid_remainder(0)),
47        ('_', valid_remainder(1)),
48    ))
49    .take()
50    .parse_next(input)
51}
52
53fn valid_remainder<'a>(
54    minimum: usize,
55) -> impl ModalParser<&'a str, &'a str, winnow::error::ContextError> {
56    move |input: &mut &'a str| {
57        take_while(
58            // Use .. instead of ..= since we've already processed a single character
59            minimum..MAX_IDENTIFIER_LENGTH,
60            (b'_', b'a'..=b'z', b'A'..=b'Z', b'0'..=b'9'),
61        )
62        .parse_next(input)
63    }
64}
65
66fn parse_address<'s>(input: &mut &'s str) -> ModalResult<&'s str> {
67    ("0x", take_while(1..=64, AsChar::is_hex_digit))
68        .take()
69        .parse_next(input)
70}

72pub(super) fn parse_type_tag(mut input: &str) -> Result<TypeTag, Option<String>> {
73    let nodes = Cell::new(0u64);
74    let tag = type_tag(&mut input, 0, &nodes).map_err(extract_message)?;
75    eof.parse_next(&mut input).map_err(extract_message)?;
76    Ok(tag)
77}

79/// Extract a human-readable message from a winnow error, if one was attached
80/// via `StrContext::Label`.
81fn extract_message(err: winnow::error::ErrMode<winnow::error::ContextError>) -> Option<String> {
82    use winnow::error::StrContext;
83
84    let inner = match err {
85        winnow::error::ErrMode::Backtrack(e) | winnow::error::ErrMode::Cut(e) => e,
86        winnow::error::ErrMode::Incomplete(_) => return None,
87    };
88
89    inner.context().find_map(|ctx| match ctx {
90        StrContext::Label(msg) => Some(msg.to_string()),
91        _ => None,
92    })
93}

95/// Increment the node counter and reject if it exceeds `MAX_TYPE_NODE_COUNT`.
96/// Called once per `TypeTag` instance (every primitive, every `vector<...>`,
97/// and every `struct<...>`); each instance counts as one node.
98fn account_node(
99    nodes: &Cell<u64>,
100) -> Result<(), winnow::error::ErrMode<winnow::error::ContextError>> {
101    let next = nodes.get().saturating_add(1);
102    if next > MAX_TYPE_NODE_COUNT {
103        let mut err = winnow::error::ContextError::new();
104        err.push(winnow::error::StrContext::Label(
105            "type exceeds the maximum node count of 256",
106        ));
107        return Err(winnow::error::ErrMode::Cut(err));
108    }
109    nodes.set(next);
110    Ok(())
111}
112
113fn type_tag(input: &mut &str, depth: usize, nodes: &Cell<u64>) -> ModalResult<TypeTag> {
114    account_node(nodes)?;
115
116    if depth >= MAX_TYPE_TAG_DEPTH {
117        let mut err = winnow::error::ContextError::new();
118        err.push(winnow::error::StrContext::Label(
119            "type exceeds the maximum nesting depth of 128",
120        ));
121        return Err(winnow::error::ErrMode::Cut(err));
122    }
123
124    alt((
125        "u8".value(TypeTag::U8),
126        "u16".value(TypeTag::U16),
127        "u32".value(TypeTag::U32),
128        "u64".value(TypeTag::U64),
129        "u128".value(TypeTag::U128),
130        "u256".value(TypeTag::U256),
131        "bool".value(TypeTag::Bool),
132        "address".value(TypeTag::Address),
133        "signer".value(TypeTag::Signer),
134        |input: &mut &str| {
135            "vector<".parse_next(input)?;
136            let ty = type_tag(input, depth + 1, nodes)?;
137            ">".parse_next(input)?;
138            Ok(TypeTag::Vector(Box::new(ty)))
139        },
140        |input: &mut &str| struct_tag(input, depth, nodes).map(|s| TypeTag::Struct(Box::new(s))),
141    ))
142    .parse_next(input)
143}

145pub(super) fn parse_struct_tag(mut input: &str) -> Result<StructTag, Option<String>> {
146    let nodes = Cell::new(0u64);
147    // Count the outer StructTag itself as one node, matching the behaviour
148    // when the same shape is parsed as `TypeTag::Struct(_)` via
149    // `parse_type_tag`. Without this, the two entry points would have
150    // different node-count boundaries for the same logical type.
151    account_node(&nodes).map_err(extract_message)?;
152    let tag = struct_tag(&mut input, 0, &nodes).map_err(extract_message)?;
153    eof.parse_next(&mut input).map_err(extract_message)?;
154    Ok(tag)
155}

157fn struct_tag(input: &mut &str, depth: usize, nodes: &Cell<u64>) -> ModalResult<StructTag> {
158    let (address, _, module, _, name) = (
159        parse_address.try_map(|s| s.parse::<Address>()),
160        "::",
161        identifier.map(|ident| Identifier(ident.into())),
162        "::",
163        identifier.map(|ident| Identifier(ident.into())),
164    )
165        .parse_next(input)?;
166
167    // optional generic
168    let generics = opt(|input: &mut &str| {
169        "<".parse_next(input)?;
170        let result = generics_with_depth(input, depth, nodes)?;
171        ">".parse_next(input)?;
172        Ok(result)
173    })
174    .parse_next(input)?
175    .unwrap_or_default();
176
177    Ok(StructTag {
178        address,
179        module,
180        name,
181        type_params: generics,
182    })
183}
184
185fn generics_with_depth(
186    input: &mut &str,
187    depth: usize,
188    nodes: &Cell<u64>,
189) -> ModalResult<Vec<TypeTag>> {
190    separated(
191        1..,
192        delimited(
193            space0,
194            |input: &mut &str| type_tag(input, depth + 1, nodes),
195            space0,
196        ),
197        ",",
198    )
199    .parse_next(input)
200}

202//
203// const identifier validity check
204//
205pub(super) const fn is_valid_identifier(s: &str) -> bool {
206    const fn is_valid_remainder_byte(b: u8) -> bool {
207        matches!(b, b'_' | b'a'..=b'z' | b'A'..=b'Z' | b'0'..=b'9')
208    }
209
210    let bytes = s.as_bytes();
211    if bytes.is_empty() || bytes.len() > MAX_IDENTIFIER_LENGTH {
212        return false;
213    }
214
215    // Check that the identifier either starts with an alpha or an underscore + at least one more
216    // byte
217    let remainder = match bytes {
218        [b'a'..=b'z', remainder @ ..] | [b'A'..=b'Z', remainder @ ..] => remainder,
219        [b'_', remainder @ ..] if !remainder.is_empty() => remainder,
220        _ => return false,
221    };
222
223    let mut i = 0;
224    while i < remainder.len() {
225        if !is_valid_remainder_byte(remainder[i]) {
226            return false;
227        }
228        i += 1;
229    }
230
231    true
232}

493    // Custom strategies that exercise the full TypeTag grammar, including vectors and nested
494    // generics. The existing `Arbitrary` derives intentionally skip these (Vector has weight 0,
495    // type_params is always empty) to keep BCS serialization fuzz tests fast.
496
497    /// Strategy producing valid Move identifiers, including both
498    /// letter-prefixed (`Foo`, `a1`) and underscore-prefixed (`_bar`, `__x1`).
499    fn arb_identifier() -> impl Strategy<Value = Identifier> {
500        prop_oneof!["[a-zA-Z][a-zA-Z0-9_]{0,127}", "_[a-zA-Z0-9_]{1,127}",]
501            .prop_map(|s| Identifier::new(&s).unwrap())
502    }
503
504    /// Strategy producing a StructTag with empty type_params.
505    fn arb_struct_tag_base() -> impl Strategy<Value = StructTag> {
506        (any::<Address>(), arb_identifier(), arb_identifier())
507            .prop_map(|(addr, module, name)| StructTag::new(addr, module, name, Vec::new()))
508    }
509
510    /// Strategy producing valid TypeTags including vectors, structs with generics,
511    /// and nested combinations. Recursion is bounded to depth 4.
512    fn arb_type_tag() -> impl Strategy<Value = TypeTag> {
513        let leaf = prop_oneof![
514            Just(TypeTag::U8),
515            Just(TypeTag::U16),
516            Just(TypeTag::U32),
517            Just(TypeTag::U64),
518            Just(TypeTag::U128),
519            Just(TypeTag::U256),
520            Just(TypeTag::Bool),
521            Just(TypeTag::Address),
522            Just(TypeTag::Signer),
523        ];
524
525        leaf.prop_recursive(
526            4,  // max depth of recursion
527            64, // desired_size (target number of nodes)
528            3,  // expected_branch_size per recursive step
529            |inner| {
530                let vector_strat = inner.clone().prop_map(|t| TypeTag::Vector(Box::new(t)));
531
532                let struct_no_params =
533                    arb_struct_tag_base().prop_map(|s| TypeTag::Struct(Box::new(s)));
534
535                let struct_with_params = (
536                    any::<Address>(),
537                    arb_identifier(),
538                    arb_identifier(),
539                    proptest::collection::vec(inner, 1..=3),
540                )
541                    .prop_map(|(addr, module, name, params)| {
542                        TypeTag::Struct(Box::new(StructTag::new(addr, module, name, params)))
543                    });
544
545                prop_oneof![vector_strat, struct_no_params, struct_with_params,]
546            },
547        )
548    }
549
550    /// Reduced set of primitive TypeTag leaves for depth-focused tests.
551    fn arb_simple_leaf() -> impl Strategy<Value = TypeTag> {
552        prop_oneof![Just(TypeTag::U8), Just(TypeTag::U64), Just(TypeTag::Bool)]
553    }
