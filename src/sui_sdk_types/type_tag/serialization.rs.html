16/// Increment a shared `Cell<u64>` node counter and reject when the running
17/// total exceeds `MAX_TYPE_NODE_COUNT`. The counter is allocated on the
18/// stack at the top-level entry point and shared by reference through
19/// every recursive seed; `Cell` provides single-threaded interior
20/// mutability without the memory-barrier overhead of an atomic, and
21/// references are `Copy` so seeds carry it without any cloning.
22fn account_node<E: serde::de::Error>(nodes: &Cell<u64>) -> Result<(), E> {
23    let next = nodes.get().saturating_add(1);
24    if next > MAX_TYPE_NODE_COUNT {
25        return Err(serde::de::Error::custom(format!(
26            "type tag exceeds max node count of {MAX_TYPE_NODE_COUNT}"
27        )));
28    }
29    nodes.set(next);
30    Ok(())
31}

33/// Seed that threads the current 0-indexed nesting depth and a borrowed
34/// node counter through the recursive BCS `TypeTag` deserialization graph.
35/// Each successive `TypeTag` boundary (inside `Vector(inner)` or
36/// `StructTag.type_params[i]`) receives a seed with `depth + 1` carrying
37/// the same counter reference. Rejects on
38/// `depth >= MAX_TYPE_TAG_DEPTH` (catches deep nesting) or
39/// `node count > MAX_TYPE_NODE_COUNT` (catches wide types), matching the
40/// bounds enforced by the string parser at `parse::type_tag`.
41struct TypeTagSeed<'a> {
42    depth: usize,
43    nodes: &'a Cell<u64>,
44}
45
46impl<'de, 'a> DeserializeSeed<'de> for TypeTagSeed<'a> {
47    type Value = TypeTag;
48
49    fn deserialize<D>(self, deserializer: D) -> Result<TypeTag, D::Error>
50    where
51        D: Deserializer<'de>,
52    {
53        account_node(self.nodes)?;
54        if self.depth >= MAX_TYPE_TAG_DEPTH {
55            return Err(serde::de::Error::custom(format!(
56                "type tag exceeds max nesting depth of {MAX_TYPE_TAG_DEPTH}"
57            )));
58        }
59        let variants = &[
60            "bool", "u8", "u64", "u128", "address", "signer", "vector", "struct", "u16", "u32",
61            "u256",
62        ];
63        deserializer.deserialize_enum(
64            "TypeTag",
65            variants,
66            TypeTagSeedVisitor {
67                depth: self.depth,
68                nodes: self.nodes,
69            },
70        )
71    }
72}

268struct TypeTagSeedVisitor<'a> {
269    depth: usize,
270    nodes: &'a Cell<u64>,
271}
272
273impl<'de, 'a> Visitor<'de> for TypeTagSeedVisitor<'a> {
274    type Value = TypeTag;
275
276    fn expecting(&self, formatter: &mut std::fmt::Formatter) -> std::fmt::Result {
277        formatter.write_str("TypeTag")
278    }
279
280    fn visit_enum<A>(self, data: A) -> Result<Self::Value, A::Error>
281    where
282        A: serde::de::EnumAccess<'de>,
283    {
284        use serde::de::VariantAccess;
285
286        let (variant, deserializer) = data.variant::<u32>()?;
287        match SerializedTypeTagVariant::new(variant)
288            .map_err(|e| serde::de::Error::custom(format!("unknown variant {e}")))?
289        {
290            SerializedTypeTagVariant::Bool => deserializer.unit_variant().map(|_| TypeTag::Bool),
291            SerializedTypeTagVariant::U8 => deserializer.unit_variant().map(|_| TypeTag::U8),
292            SerializedTypeTagVariant::U64 => deserializer.unit_variant().map(|_| TypeTag::U64),
293            SerializedTypeTagVariant::U128 => deserializer.unit_variant().map(|_| TypeTag::U128),
294            SerializedTypeTagVariant::Address => {
295                deserializer.unit_variant().map(|_| TypeTag::Address)
296            }
297            SerializedTypeTagVariant::Signer => {
298                deserializer.unit_variant().map(|_| TypeTag::Signer)
299            }
300            SerializedTypeTagVariant::Vector => deserializer
301                .newtype_variant_seed(TypeTagSeed {
302                    depth: self.depth + 1,
303                    nodes: self.nodes,
304                })
305                .map(|t| TypeTag::Vector(Box::new(t))),
306            SerializedTypeTagVariant::Struct => deserializer
307                .newtype_variant_seed(StructTagSeed {
308                    type_param_depth: self.depth + 1,
309                    nodes: self.nodes,
310                })
311                .map(|s| TypeTag::Struct(Box::new(s))),
312            SerializedTypeTagVariant::U16 => deserializer.unit_variant().map(|_| TypeTag::U16),
313            SerializedTypeTagVariant::U32 => deserializer.unit_variant().map(|_| TypeTag::U32),
314            SerializedTypeTagVariant::U256 => deserializer.unit_variant().map(|_| TypeTag::U256),
315        }
316    }
317}

319struct StructTagSeedVisitor<'a> {
320    type_param_depth: usize,
321    nodes: &'a Cell<u64>,
322}
323
324impl<'de, 'a> Visitor<'de> for StructTagSeedVisitor<'a> {
325    type Value = StructTag;
326
327    fn expecting(&self, formatter: &mut std::fmt::Formatter) -> std::fmt::Result {
328        formatter.write_str("a StructTag")
329    }
330
331    fn visit_seq<A>(self, mut seq: A) -> Result<Self::Value, A::Error>
332    where
333        A: serde::de::SeqAccess<'de>,
334    {
335        let address: Address = seq
336            .next_element()?
337            .ok_or_else(|| serde::de::Error::custom("missing StructTag address"))?;
338        let module: Identifier = seq
339            .next_element()?
340            .ok_or_else(|| serde::de::Error::custom("missing StructTag module"))?;
341        let name: Identifier = seq
342            .next_element()?
343            .ok_or_else(|| serde::de::Error::custom("missing StructTag name"))?;
344        let type_params: Vec<TypeTag> = seq
345            .next_element_seed(TypeTagVecSeed {
346                depth: self.type_param_depth,
347                nodes: self.nodes,
348            })?
349            .ok_or_else(|| serde::de::Error::custom("missing StructTag type_params"))?;
350
351        Ok(StructTag {
352            address,
353            module,
354            name,
355            type_params,
356        })
357    }
358}

360struct TypeTagVecVisitor<'a> {
361    depth: usize,
362    nodes: &'a Cell<u64>,
363}
364
365impl<'de, 'a> Visitor<'de> for TypeTagVecVisitor<'a> {
366    type Value = Vec<TypeTag>;
367
368    fn expecting(&self, formatter: &mut std::fmt::Formatter) -> std::fmt::Result {
369        formatter.write_str("a sequence of TypeTags")
370    }
371
372    fn visit_seq<A>(self, mut seq: A) -> Result<Self::Value, A::Error>
373    where
374        A: serde::de::SeqAccess<'de>,
375    {
376        // Clamp the wire-supplied length hint by the remaining node budget
377        // so an attacker-supplied uleb128 sequence length cannot drive
378        // `Vec::with_capacity` past the node-count bound the deserializer
379        // is already prepared to enforce. Without this clamp, the bcs
380        // crate's `SeqAccess::size_hint` propagates the parsed uleb32
381        // length (up to ~2 billion) and `Vec::<TypeTag>::with_capacity`
382        // attempts a multi-GB allocation before any element is parsed.
383        let remaining = MAX_TYPE_NODE_COUNT.saturating_sub(self.nodes.get()) as usize;
384        let cap = seq.size_hint().unwrap_or(0).min(remaining);
385        let mut out = Vec::with_capacity(cap);
386        while let Some(t) = seq.next_element_seed(TypeTagSeed {
387            depth: self.depth,
388            nodes: self.nodes,
389        })? {
390            out.push(t);
391        }
392        Ok(out)
393    }
394}

397struct BinaryStructTagRef<'a> {
398    address: &'a Address,
399    module: &'a Identifier,
400    name: &'a Identifier,
401    type_params: &'a [TypeTag],
402}
403
404impl Serialize for StructTag {
405    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
406    where
407        S: Serializer,
408    {
409        if serializer.is_human_readable() {
410            serde_with::DisplayFromStr::serialize_as(self, serializer)
411        } else {
412            BinaryStructTagRef {
413                address: &self.address,
414                module: &self.module,
415                name: &self.name,
416                type_params: &self.type_params,
417            }
418            .serialize(serializer)
419        }
420    }
421}
422
423impl<'de> Deserialize<'de> for StructTag {
424    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
425    where
426        D: Deserializer<'de>,
427    {
428        if deserializer.is_human_readable() {
429            serde_with::DisplayFromStr::deserialize_as(deserializer)
430        } else {
431            // Treat the `StructTag` as if it were the body of a TypeTag at
432            // depth 0, so its `type_params` start at depth 1 — matching
433            // `parse_struct_tag`, which calls `struct_tag(depth=0)` →
434            // `generics_with_depth(depth=0)` → `type_tag(depth+1=1)`.
435            // Also count the outer StructTag as one node so the
436            // `bcs::from_bytes::<StructTag>` boundary matches
437            // `bcs::from_bytes::<TypeTag>` for the same shape.
438            let nodes = Cell::new(0);
439            account_node(&nodes)?;
440            StructTagSeed {
441                type_param_depth: 1,
442                nodes: &nodes,
443            }
444            .deserialize(deserializer)
445        }
446    }

628    fn bcs_struct_tag_at_max_node_count_is_rejected() {
629        let st = StructTag {
630            address: Address::from_str("0x1").unwrap(),
631            module: Identifier("M".into()),
632            name: Identifier("F".into()),
633            type_params: vec![TypeTag::U8; MAX_TYPE_NODE_COUNT as usize],
634        };
635        let bytes = bcs::to_bytes(&st).unwrap();
636        let err = bcs::from_bytes::<StructTag>(&bytes).unwrap_err();
637        assert!(
638            err.to_string().contains("node count"),
639            "expected node-count error, got: {err}"
640        );
641    }

643    #[test]
644    fn bcs_struct_tag_at_exact_max_node_count_is_accepted() {
645        // `MAX - 1` u8 type_params + the outer StructTag = exactly MAX
646        // nodes.
647        let st = StructTag {
648            address: Address::from_str("0x1").unwrap(),
649            module: Identifier("M".into()),
650            name: Identifier("F".into()),
651            type_params: vec![TypeTag::U8; (MAX_TYPE_NODE_COUNT - 1) as usize],
652        };
653        let bytes = bcs::to_bytes(&st).unwrap();
654        let parsed: StructTag = bcs::from_bytes(&bytes).unwrap();

658    // Regression test: the visitor used to call
659    // `Vec::with_capacity(seq.size_hint().unwrap_or(0))` before any
660    // per-element bound was enforced, so an attacker-supplied uleb32
661    // type_params length of `0x7fffffff` could request a ~32 GB
662    // allocation from a 42-byte BCS payload. The visitor must now clamp
663    // the hint by the remaining node-count budget so this payload fails
664    // through normal short-input handling instead of aborting on
665    // allocation failure. If the regression returns, this test will
666    // OOM-abort the process before it can fail cleanly.
667    #[test]
668    fn bcs_struct_tag_huge_type_params_length_does_not_overallocate() {
669        // Variant tag 0x07 (Struct), 32-byte zero address, "a" as
670        // module, "b" as name, then a uleb128 of 0x7fffffff for the
671        // length of `type_params`.
672        let mut bytes = vec![SerializedTypeTagVariant::Struct as u8];
673        bytes.extend_from_slice(&[0u8; Address::LENGTH]);
674        bytes.extend_from_slice(&[0x01, b'a']);
675        bytes.extend_from_slice(&[0x01, b'b']);
676        bytes.extend_from_slice(&[0xff, 0xff, 0xff, 0xff, 0x07]);
677
678        assert!(bcs::from_bytes::<TypeTag>(&bytes).is_err());
679    }
