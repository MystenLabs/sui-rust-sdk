32fn sui_binary() -> &'static Path {
33    static SUI_BINARY: std::sync::OnceLock<PathBuf> = std::sync::OnceLock::new();
34
35    SUI_BINARY
36        .get_or_init(|| {
37            if let Ok(path) = std::env::var("SUI_BINARY") {
38                return PathBuf::from(path);
39            }
40            if let Ok(output) = Command::new("which").arg("sui").output()
41                && output.status.success()
42            {
43                let path = String::from_utf8_lossy(&output.stdout).trim().to_string();
44                if !path.is_empty() {
45                    return PathBuf::from(path);
46                }
47            }
48            panic!("sui binary not found. Please install sui or set SUI_BINARY env var")
49        })
50        .as_path()
51}

53async fn wait_for_ready(client: &mut Client) -> Result<()> {
54    // Wait till the network has started up and at least one checkpoint has been produced
55    for _ in 0..NETWORK_STARTUP_TIMEOUT_SECS {
56        if let Ok(resp) = client
57            .ledger_client()
58            .get_service_info(sui_rpc::proto::sui::rpc::v2::GetServiceInfoRequest::default())
59            .await
60            && resp.into_inner().checkpoint_height() > 0
61        {
62            return Ok(());
63        }
64        sleep(Duration::from_secs(NETWORK_STARTUP_POLL_INTERVAL_SECS)).await;
65    }
66    anyhow::bail!(
67        "Network failed to start within {}s timeout",
68        NETWORK_STARTUP_TIMEOUT_SECS,
69    )
70}

137    pub async fn build(self) -> Result<SuiNetworkHandle> {
138        let dir = TempDir::new()?;
139        self.generate_genesis(dir.path())?;
140        let (validator_keys, user_keys) = load_keys(dir.path())?;
141
142        let rpc_port = get_available_port();
143        let process = self.start_network(dir.path(), rpc_port)?;
144
145        let rpc_url = format!("http://127.0.0.1:{rpc_port}");
146
147        let mut client = sui_rpc::Client::new(&rpc_url)?;
148        wait_for_ready(&mut client).await?;
149        let mut sui = SuiNetworkHandle {
150            process,
151            dir,
152            rpc_url,
153            client,
154            num_validators: self.num_validators,
155            epoch_duration_ms: self.epoch_duration_ms,
156            validator_keys,
157            user_keys,
158        };
159
160        // Make sure SuiSystemState has been upgraded to v2
161        sui.upgrade_sui_system_state().await?;
162
163        // Make sure validator accounts are funded
164        let fund_requests = sui
165            .validator_keys
166            .keys()
167            // give each validator 1M SUI
168            .map(|address| (*address, 1_000_000 * 1_000_000_000))
169            .collect::<Vec<_>>();
170        sui.fund(&fund_requests).await?;
171
172        Ok(sui)
173    }

176        std::fs::create_dir_all(dir)?;
177        let mut cmd = Command::new(sui_binary());
178        cmd.arg("genesis")
179            .arg("--working-dir")
180            .arg(dir)

183            .arg("--committee-size")
184            .arg(self.num_validators.to_string())
185            .arg("--with-faucet");
186        let status = cmd.status()?;
187        if !status.success() {
188            return Err(anyhow::anyhow!("Failed to generate genesis"));
189        }
190        Ok(())
191    }

193    fn start_network(&self, dir: &Path, rpc_port: u16) -> Result<Child> {
194        let stdout_name = dir.join("out.stdout");
195        let stdout = std::fs::File::create(stdout_name)?;
196        let stderr_name = dir.join("out.stderr");
197        let stderr = std::fs::File::create(stderr_name)?;
198
199        let mut cmd = Command::new(sui_binary());
200
201        cmd.arg("start")
202            .arg("--network.config")
203            .arg(dir)
204            .arg("--fullnode-rpc-port")
205            .arg(rpc_port.to_string())
206            .stdout(stdout)
207            .stderr(stderr)
208            .spawn()
209            .map_err(|e| anyhow!("Failed to run `sui start`: {e}"))
210    }

213fn keypair_from_base64(b64: &str) -> Result<Ed25519PrivateKey> {
214    let bytes = <base64ct::Base64 as base64ct::Encoding>::decode_vec(b64)?;
215
216    let keypair =
217        match SignatureScheme::from_byte(*bytes.first().ok_or_else(|| anyhow!("Invalid key"))?)
218            .map_err(|e| anyhow!("{e}"))?
219        {
220            SignatureScheme::Ed25519 => Ed25519PrivateKey::new(
221                bytes
222                    .get(1..)
223                    .ok_or_else(|| anyhow!("Invalid key"))?
224                    .try_into()?,
225            ),
226            SignatureScheme::Secp256k1 => bail!("invalid key"),
227            SignatureScheme::Secp256r1 => bail!("invalid key"),
228            _ => bail!("invalid key"),
229        };
230
231    Ok(keypair)
232}

234fn ed25519_private_key_from_base64(b64: &str) -> Result<Ed25519PrivateKey> {
235    let bytes = <base64ct::Base64 as base64ct::Encoding>::decode_vec(b64)?;
236    Ok(Ed25519PrivateKey::new((&bytes[..]).try_into()?))
237}

239fn load_keys(dir: &Path) -> Result<(BTreeMap<Address, Ed25519PrivateKey>, Vec<Ed25519PrivateKey>)> {
240    #[derive(serde::Deserialize)]
241    struct Config {
242        validator_configs: Vec<NodeConfig>,
243        account_keys: Vec<String>,
244    }
245
246    #[derive(serde::Deserialize)]
247    #[serde(rename_all = "kebab-case")]
248    struct NodeConfig {
249        account_key_pair: RawKey,
250    }
251
252    #[derive(serde::Deserialize)]
253    struct RawKey {
254        value: String,
255    }
256
257    let raw = std::fs::read(dir.join("network.yaml"))?;
258    let network_config: Config = serde_yaml::from_slice(&raw)?;
259
260    let mut validator_keys = BTreeMap::new();
261
262    for validator in network_config.validator_configs {
263        let keypair = keypair_from_base64(&validator.account_key_pair.value)?;
264        let address = keypair.public_key().derive_address();
265        validator_keys.insert(address, keypair);
266    }
267
268    let mut user_keys = vec![];
269
270    for raw_key in network_config.account_keys {
271        user_keys.push(ed25519_private_key_from_base64(&raw_key)?);
272    }
273
274    Ok((validator_keys, user_keys))
275}

277impl SuiNetworkHandle {
278    pub async fn fund(&mut self, requests: &[(Address, u64)]) -> Result<()> {
279        let private_key = self.user_keys.first().unwrap();
280        let sender = private_key.public_key().derive_address();
281
282        let mut builder = TransactionBuilder::new();
283        builder.set_sender(sender);
284
285        for (address, amount) in requests {
286            let recipient = builder.pure(address);
287            let coin = builder.intent(CoinWithBalance::sui(*amount));
288            builder.transfer_objects(vec![coin], recipient);
289        }
290
291        let transaction = builder.build(&mut self.client).await?;
292
293        let signature = private_key.sign_transaction(&transaction)?;
294
295        let response = self
296            .client
297            .execute_transaction_and_wait_for_checkpoint(
298                ExecuteTransactionRequest::new(transaction.into())
299                    .with_signatures(vec![signature.into()])
300                    .with_read_mask(FieldMask::from_str("*")),
301                std::time::Duration::from_secs(10),
302            )
303            .await?
304            .into_inner();
305
306        assert!(
307            response.transaction().effects().status().success(),
308            "fund failed"
309        );
310        Ok(())
311    }

351    pub fn build_package(&self, package: &Path) -> Result<(sui_sdk_types::Publish, Digest)> {
352        #[derive(serde_derive::Deserialize)]
353        struct MoveBuildOutput {
354            modules: Vec<String>,
355            dependencies: Vec<Address>,
356            digest: Vec<u8>,
357        }
358        let client_config = self.dir.path().join("client.yaml");
359
360        let mut cmd = Command::new(sui_binary());
361        cmd.arg("move")
362            .arg("--client.config")
363            .arg(client_config)
364            .arg("-p")
365            .arg(package)
366            .arg("build")
367            .arg("--ignore-chain") // TODO remove once 1.62 is released
368            .arg("--dump-bytecode-as-base64");
369        let output = cmd.output()?;
370
371        if !output.status.success() {
372            return Err(anyhow::anyhow!(
373                "stdout: {}\n\n stderr: {}",
374                output.stdout.escape_ascii(),
375                output.stderr.escape_ascii()
376            ));
377        }
378
379        let move_build_output: MoveBuildOutput = serde_json::from_slice(&output.stdout)?;
380        let modules = move_build_output
381            .modules
382            .into_iter()
383            .map(|b64| <base64ct::Base64 as base64ct::Encoding>::decode_vec(&b64))
384            .collect::<Result<Vec<_>, _>>()?;
385        let digest = Digest::from_bytes(move_build_output.digest)?;
386
387        Ok((
388            sui_sdk_types::Publish {
389                modules,
390                dependencies: move_build_output.dependencies,
391            },
392            digest,
393        ))
394    }

397/// Return an ephemeral, available port. On unix systems, the port returned will be in the
398/// TIME_WAIT state ensuring that the OS won't hand out this port for some grace period.
399/// Callers should be able to bind to this port given they use SO_REUSEADDR.
400fn get_available_port() -> u16 {
401    const MAX_PORT_RETRIES: u32 = 1000;
402
403    for _ in 0..MAX_PORT_RETRIES {
404        if let Ok(port) = get_ephemeral_port() {
405            return port;
406        }
407    }
408
409    panic!("Error: could not find an available port on localhost");
410}
411
412fn get_ephemeral_port() -> std::io::Result<u16> {
413    use std::net::SocketAddr;
414    use std::net::TcpListener;
415    use std::net::TcpStream;
416
417    // Request a random available port from the OS
418    let listener = TcpListener::bind(SocketAddr::from(([127, 0, 0, 1], 0)))?;
419    let addr = listener.local_addr()?;
420
421    // Create and accept a connection (which we'll promptly drop) in order to force the port
422    // into the TIME_WAIT state, ensuring that the port will be reserved from some limited
423    // amount of time (roughly 60s on some Linux systems)
424    let _sender = TcpStream::connect(addr)?;
425    let _incoming = listener.accept()?;
426
427    Ok(addr.port())
428}

439    #[tokio::test]
440    async fn it_works() -> Result<(), anyhow::Error> {
441        let mut sui = SuiNetworkBuilder::default().build().await?;
442
443        // stream ~10 checkpoints to make sure things work
444        let mut stream = sui
445            .client
446            .subscription_client()
447            .subscribe_checkpoints(
448                SubscribeCheckpointsRequest::default()
449                    .with_read_mask(FieldMask::from_str("sequence_number")),
450            )
451            .await?
452            .into_inner();
453
454        let mut count = 0;
455        let mut last = None;
456        while let Some(item) = stream.next().await {
457            let checkpoint = item.unwrap();
458            let cursor = checkpoint.cursor.unwrap();
459            assert_eq!(
460                cursor,
461                checkpoint.checkpoint.unwrap().sequence_number.unwrap()
462            );
463            println!("checkpoint: {cursor}");
464
465            if let Some(last) = last {
466                assert_eq!(last, cursor - 1);
467            }
468            last = Some(cursor);
469
470            // Subscribe for 10 checkponts to ensure the subscription system works
471            count += 1;
472            if count > 10 {
473                break;
474            }
475        }
476
477        assert!(count >= 10);
478
479        Ok(())
480    }
