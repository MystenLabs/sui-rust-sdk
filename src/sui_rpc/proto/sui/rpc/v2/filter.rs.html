4//! Ergonomic constructors for the transaction and event filter DNF types.
5//!
6//! The generated [`TransactionFilter`] and [`EventFilter`] messages model a
7//! filter in disjunctive normal form: a filter is an OR of terms, each term is
8//! an AND of literals, and each literal is a predicate that may be negated.
9//! Building even the simplest filter by hand therefore means nesting three
10//! levels of message plus a single-field predicate wrapper. The helpers in this
11//! module collapse that nesting so the common cases read as one expression.
12//!
13//! The predicate constructors live in the [`transaction`] and [`event`]
14//! submodules (one function per predicate kind) because the generated messages
15//! already expose the natural names -- `sender`, `affected_object`, and so on
16//! -- as field accessors. Each constructor returns a plain, un-negated literal,
17//! so a term whose literals are all positive can be written from bare
18//! constructor calls; call [`TransactionLiteral::negate`] (or the event
19//! equivalent) to require instead that a predicate not match.
20//!
21//! # Examples
22//!
23//! A transaction filter, from the simplest predicate up to a full DNF
24//! expression:
25//!
26//! ```
27//! use sui_rpc::proto::sui::rpc::v2::TransactionFilter;
28//! use sui_rpc::proto::sui::rpc::v2::TransactionTerm;
29//! use sui_rpc::proto::sui::rpc::v2::filter::transaction as tx;
30//! use sui_sdk_types::Address;
31//!
32//! let a = Address::ZERO;
33//! let o = Address::TWO;
34//!
35//! // Match transactions sent by `a`.
36//! let _ = TransactionFilter::matching(tx::sender(a));
37//!
38//! // Match transactions that were sent by `a` and also touched object `o`.
39//! let _ = TransactionFilter::all([tx::sender(a), tx::affected_object(o)]);
40//!
41//! // Match `(sender = a AND NOT package_write) OR (affected_object = o)`.
42//! let _ = TransactionFilter::any([
43//!     TransactionTerm::all([tx::sender(a), tx::package_write().negate()]),
44//!     TransactionTerm::all([tx::affected_object(o)]),
45//! ]);
46//! ```
47//!
48//! Event filters use the [`event`] submodule and the same combinators:
49//!
50//! ```
51//! use sui_rpc::proto::sui::rpc::v2::EventFilter;
52//! use sui_rpc::proto::sui::rpc::v2::filter::event as ev;
53//! use sui_sdk_types::Address;
54//! use sui_sdk_types::Identifier;
55//!
56//! let a = Address::ZERO;
57//! let coin_module: Identifier = "coin".parse().unwrap();
58//!
59//! // Match events emitted by transactions sent by `a` from the `0x2::coin`
60//! // module.
61//! let _ = EventFilter::all([ev::sender(a), ev::emit_module((Address::TWO, coin_module))]);
62//! ```
63//!
64//! Move-path and type predicates take typed components as readily as strings,
65//! so callers never have to hand-format a `::`-delimited path:
66//!
67//! ```
68//! use sui_rpc::proto::sui::rpc::v2::filter::transaction as tx;
69//! use sui_sdk_types::Address;
70//! use sui_sdk_types::Identifier;
71//! use sui_sdk_types::StructTag;
72//!
73//! let package = Address::TWO;
74//! let module: Identifier = "coin".parse().unwrap();
75//! let function: Identifier = "mint".parse().unwrap();
76//!
77//! // Each of these builds a Move-call predicate at a different specificity.
78//! let _ = tx::move_call(package); // whole package
79//! let _ = tx::move_call((package, module.clone())); // package::module
80//! let _ = tx::move_call((package, module, function)); // exact function
81//! let _ = tx::move_call("0x2::coin::mint"); // or a raw path string
82//!
83//! // Event-type predicates accept a `StructTag`, a `TypeTag`, or a string.
84//! let coin = StructTag::new(
85//!     package,
86//!     "coin".parse().unwrap(),
87//!     "Coin".parse().unwrap(),
88//!     Vec::new(),
89//! );
90//! let _ = tx::event_type(coin);
91//! let _ = tx::event_type("0x2::coin::Coin");
92//! ```

205// Layer 3 (transaction): negation on literals and the term/filter combinators.
206// A predicate constructor already yields an un-negated literal, and a literal
207// converts into a single-literal term, so the all-positive cases need no
208// explicit wrapping.
209
210impl TransactionLiteral {
211    /// Negate this literal: require that its predicate not match.
212    pub fn negate(self) -> Self {
213        self.with_negated(true)
214    }
215}
216
217impl From<TransactionLiteral> for TransactionTerm {
218    fn from(literal: TransactionLiteral) -> Self {
219        Self::all([literal])
220    }
221}
222
223impl TransactionTerm {
224    /// A term matching transactions that satisfy every one of `literals`
225    /// (the literals are ANDed together).
226    pub fn all(literals: impl IntoIterator<Item = impl Into<TransactionLiteral>>) -> Self {
227        Self::default().with_literals(literals.into_iter().map(Into::into).collect())
228    }
229}
230
231impl TransactionFilter {
232    /// A filter matching transactions that satisfy any one of `terms`
233    /// (the terms are ORed together).
234    pub fn any(terms: impl IntoIterator<Item = impl Into<TransactionTerm>>) -> Self {
235        Self::default().with_terms(terms.into_iter().map(Into::into).collect())
236    }
237
238    /// A filter with a single term matching transactions that satisfy every
239    /// one of `literals`.
240    pub fn all(literals: impl IntoIterator<Item = impl Into<TransactionLiteral>>) -> Self {
241        Self::any([TransactionTerm::all(literals)])
242    }
243
244    /// A filter matching transactions that satisfy a single literal.
245    pub fn matching(literal: impl Into<TransactionLiteral>) -> Self {
246        Self::all([literal])
247    }
248}

292// Layer 2: predicate constructors. These are free functions rather than
293// associated functions because the generated messages already use the same
294// names for their field accessors.

296/// Predicate constructors for [`TransactionFilter`].
297pub mod transaction {
298    use super::super::*;
299    use sui_sdk_types::Address;
300
301    /// Match transactions sent by the specified address.
302    pub fn sender(address: impl Into<Address>) -> TransactionLiteral {
303        TransactionLiteral::default().with_sender(address.into())
304    }
305
306    /// Match transactions where the specified address's state moved as a side
307    /// effect.
308    pub fn affected_address(address: impl Into<Address>) -> TransactionLiteral {
309        TransactionLiteral::default().with_affected_address(address.into())
310    }
311
312    /// Match transactions whose effects include a change for the specified
313    /// object.
314    pub fn affected_object(object_id: impl Into<Address>) -> TransactionLiteral {
315        TransactionLiteral::default().with_affected_object(object_id.into())
316    }
317
318    /// Match transactions that made a Move call matching the specified
319    /// `package[::module[::function]]` path. Accepts a `::`-delimited path
320    /// string, an `Address` (package), an `(Address, Identifier)` package and
321    /// module pair, or an `(Address, Identifier, Identifier)` package, module,
322    /// and function triple.
323    pub fn move_call(function: impl Into<MoveCallFilter>) -> TransactionLiteral {
324        TransactionLiteral::default().with_move_call(function)
325    }
326
327    /// Match transactions that emitted an event whose package/module fields
328    /// match the specified `package[::module]` path. Accepts a `::`-delimited
329    /// path string, an `Address` (package), or an `(Address, Identifier)`
330    /// package and module pair.
331    pub fn emit_module(module: impl Into<EmitModuleFilter>) -> TransactionLiteral {
332        TransactionLiteral::default().with_emit_module(module)
333    }
334
335    /// Match transactions that emitted an event whose type matches the
336    /// specified Move type. Accepts a canonical type string, a `StructTag`, or
337    /// a `TypeTag`.
338    pub fn event_type(event_type: impl Into<EventTypeFilter>) -> TransactionLiteral {
339        TransactionLiteral::default().with_event_type(event_type)
340    }
341
342    /// Match transactions that wrote to the specified authenticated event
343    /// stream head.
344    pub fn event_stream_head(stream_id: impl Into<Address>) -> TransactionLiteral {
345        TransactionLiteral::default().with_event_stream_head(stream_id.into())
346    }
347
348    /// Match transactions that wrote a Move package -- a first publish or an
349    /// upgrade of any package.
350    pub fn package_write() -> TransactionLiteral {
351        TransactionLiteral::default().with_package_write(PackageWriteFilter::default())
352    }
353}

382    }
383}

391    #[test]
392    fn matching_builds_single_predicate_transaction_filter() {
393        let address = Address::ZERO;
394        let built = TransactionFilter::matching(transaction::sender(address));
395
396        let expected = TransactionFilter {
397            terms: vec![TransactionTerm {
398                literals: vec![TransactionLiteral {
399                    negated: false,
400                    predicate: Some(transaction_literal::Predicate::Sender(SenderFilter {
401                        address: Some(address.to_string()),
402                    })),
403                }],
404            }],
405        };
406
407        assert_eq!(built, expected);
408    }

410    #[test]
411    fn all_ands_included_predicates_into_one_term() {
412        let sender = Address::ZERO;
413        let object = Address::TWO;
414        let built = TransactionFilter::all([
415            transaction::sender(sender),
416            transaction::affected_object(object),
417        ]);
418
419        assert_eq!(built.terms.len(), 1);
420        let literals = &built.terms[0].literals;
421        assert_eq!(literals.len(), 2);
422        assert!(
423            literals
424                .iter()
425                .all(|literal| !literal.negated && literal.predicate.is_some())
426        );
427    }

429    #[test]
430    fn any_ors_terms_and_negate_sets_negated() {
431        let sender = Address::ZERO;
432        let object = Address::TWO;
433        let built = TransactionFilter::any([
434            TransactionTerm::all([
435                transaction::sender(sender),
436                transaction::package_write().negate(),
437            ]),
438            TransactionTerm::all([transaction::affected_object(object)]),
439        ]);
440
441        assert_eq!(built.terms.len(), 2);
442        assert!(!built.terms[0].literals[0].negated);
443        assert!(built.terms[0].literals[1].negated);
444    }
