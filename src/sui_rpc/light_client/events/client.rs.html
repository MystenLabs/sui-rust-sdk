33/// A streaming verifier for a single authenticated event stream.
34///
35/// Construct with [`AuthenticatedEventsClient::new`], passing a
36/// configured [`LightClient`] and an [`AuthenticatedEventsConfig`].
37/// Call [`Self::stream`] to spawn the background verifier task and
38/// receive an async stream of cryptographically-authenticated events.
39///
40/// Once [`Self::stream`] is called the client is consumed. The
41/// returned stream pulls events from the spawned task via a bounded
42/// channel; events are yielded only after the periodic reconciliation
43/// confirms the local MMR matches the on-chain
44/// [`EventStreamHead`](sui_sdk_types::framework::EventStreamHead).
45pub struct AuthenticatedEventsClient {
46    light: LightClient,
47    config: AuthenticatedEventsConfig,
48}
49
50impl AuthenticatedEventsClient {
51    /// Construct a new streaming client.
52    pub fn new(light: LightClient, config: AuthenticatedEventsConfig) -> Self {
53        Self { light, config }
54    }
55
56    /// Spawn the verifier task and return a stream of authenticated
57    /// events.
58    ///
59    /// The stream terminates when the spawned task exits — either
60    /// because the consumer dropped the receiver, or because the task
61    /// hit an unrecoverable error (the error is yielded as the last
62    /// item before termination).
63    pub fn stream(
64        self,
65    ) -> impl Stream<Item = Result<AuthenticatedEvent, LightClientError>> + Send + 'static {
66        let (tx, rx) = mpsc::channel(self.config.channel_capacity);
67        tokio::spawn(run_stream_task(self.light, self.config, tx));
68        futures::stream::unfold(rx, |mut rx| async move { rx.recv().await.map(|v| (v, rx)) })
69    }
70}

72/// Driver loop for a single stream subscription.
73///
74/// On startup: fetch the latest checkpoint as the race-free floor,
75/// then fetch the on-chain stream head. If the head exists, resume
76/// from `head.checkpoint + 1`; otherwise start from
77/// `latest_checkpoint + 1`. Either way the floor ensures no event can
78/// land in the interval without being picked up by the head check.
79///
80/// Steady state: page through `ListEvents`, fold each page into the
81/// local MMR, and at each [`AuthenticatedEventsConfig::head_check_interval`]
82/// tick fetch the on-chain head and reconcile. On match, drain the
83/// confirmed events through the channel. On mismatch, send the error
84/// and exit.
85async fn run_stream_task(
86    mut light: LightClient,
87    config: AuthenticatedEventsConfig,
88    tx: mpsc::Sender<Result<AuthenticatedEvent, LightClientError>>,
89) {
90    // The two startup helpers can fail; route the error through the
91    // channel and exit cleanly so the consumer sees one final item.
92    let start = match initial_state(&mut light, &config).await {
93        Ok(start) => start,
94        Err(e) => {
95            let _ = tx.send(Err(e)).await;
96            return;
97        }
98    };
99
100    let mut state = StreamState::new(start.initial_head, start.start_checkpoint);
101    let mut next_checkpoint = start
102        .start_checkpoint
103        .checked_add(1)
104        .unwrap_or(start.start_checkpoint);
105    let mut next_cursor: Option<prost::bytes::Bytes> = None;
106    let mut consecutive_failures = 0u32;
107    let mut last_head_check = std::time::Instant::now();
108
109    let stream_head_object_id = derive_event_stream_head_object_id(config.stream_id);
110    let filter = build_filter(&config.stream_id.to_string());
111
112    loop {
113        // Decide whether to fetch the next page or reconcile.
114        let should_reconcile = last_head_check.elapsed() >= config.head_check_interval
115            || !state.pending.is_empty()
116                && next_cursor.is_none()
117                && page_drain_done(&state, next_checkpoint);
118
119        if should_reconcile {
120            match reconcile_once(&mut light, &mut state, &stream_head_object_id).await {
121                Ok(released) => {
122                    consecutive_failures = 0;
123                    last_head_check = std::time::Instant::now();
124                    for ev in released {
125                        if tx.send(Ok(ev)).await.is_err() {
126                            return;
127                        }
128                    }
129                }
130                Err(e) if e_is_retryable(&e) => {
131                    if !backoff_or_give_up(&tx, &config, &mut consecutive_failures, e).await {
132                        return;
133                    }
134                }
135                Err(e) => {
136                    let _ = tx.send(Err(e)).await;
137                    return;
138                }
139            }
140            continue;
141        }
142
143        // Fetch the next page of events.
144        let request = ListEventsRequest {
145            read_mask: None,
146            start_checkpoint: Some(next_checkpoint),
147            end_checkpoint: None,
148            filter: Some(filter.clone()),
149            options: Some(QueryOptions {
150                limit_items: Some(config.page_size),
151                after: next_cursor.clone(),
152                before: None,
153                ordering: 0, // ascending (default)
154            }),
155        };
156
157        match fetch_one_page(&mut light, request).await {
158            Ok(page) => {
159                consecutive_failures = 0;
160                let PageResult {
161                    events,
162                    end_cursor,
163                    end_reason,
164                } = page;
165
166                if let Err(e) = process_response_batch(&mut state, events) {
167                    let _ = tx.send(Err(e)).await;
168                    return;
169                }
170
171                match end_reason {
172                    Some(QueryEndReason::ItemLimit) => {
173                        next_cursor = end_cursor;
174                    }
175                    _ => {
176                        // Server signalled tip / scan-limit / range bound.
177                        // Reset cursor and bump start checkpoint to one past
178                        // the latest fold so the next page picks up new
179                        // events as they arrive.
180                        next_cursor = None;
181                        next_checkpoint = state
182                            .local_head
183                            .checkpoint_seq
184                            .checked_add(1)
185                            .unwrap_or(next_checkpoint);
186                        // Sleep briefly so we don't spin when at the tip.
187                        tokio::time::sleep(config.retry_backoff).await;
188                    }
189                }
190            }
191            Err(e) if e_is_retryable(&e) => {
192                if !backoff_or_give_up(&tx, &config, &mut consecutive_failures, e).await {
193                    return;
194                }
195            }
196            Err(e) => {
197                let _ = tx.send(Err(e)).await;
198                return;
199            }
200        }
201    }
202}

204struct InitialState {
205    initial_head: EventStreamHead,
206    start_checkpoint: u64,
207}

209/// Establish the starting position. Fetch the latest checkpoint
210/// *before* the head so that if an event lands in between, the head
211/// fetch observes it and we resume from `head.checkpoint + 1` rather
212/// than the now-stale tip.
213///
214/// Known limitation: an `EventStreamHead` object that exists on chain
215/// but was *not modified* in the checkpoint we query produces an
216/// authenticated `NonInclusion` result, which this function treats the
217/// same as "no head yet" — it falls back to the tip. The OCS-per-
218/// checkpoint commitment scheme attests only to objects modified at a
219/// given checkpoint, not to the contents of unmodified objects, so we
220/// cannot distinguish the two cases through this API alone. In
221/// practice this is rarely an issue because new events on a stream
222/// modify its head, so a head that hasn't been touched at the tip
223/// means the stream has been quiet — and starting from the tip is the
224/// correct resume point in both cases.
225async fn initial_state(
226    light: &mut LightClient,
227    config: &AuthenticatedEventsConfig,
228) -> Result<InitialState, LightClientError> {
229    let latest_tip = light.latest_checkpoint_seq().await?;
230    let stream_head_object_id = derive_event_stream_head_object_id(config.stream_id);
231
232    let proof = light
233        .prove_object_at_checkpoint(&stream_head_object_id, latest_tip)
234        .await?;
235
236    let (initial_head, start_checkpoint) = match proof {
237        CheckpointObjectProof::Inclusion {
238            object: Some(object),
239            ..
240        } => {
241            let head = extract_event_stream_head(&object)?;
242            let cp = head.checkpoint_seq;
243            (head, cp)
244        }
245        CheckpointObjectProof::Inclusion { object: None, .. } => {
246            // The head object was deleted or wrapped at the tip.
247            // For an authenticated event stream this is unrecoverable
248            // — the on-chain anchor is gone, so no future reconciliation
249            // can succeed.
250            return Err(LightClientError::UnexpectedObjectShape {
251                reason: "event stream head was deleted or wrapped at the initial tip",
252            });
253        }
254        CheckpointObjectProof::NonInclusion => (
255            EventStreamHead::default(),
256            config.start_checkpoint.unwrap_or(latest_tip),
257        ),
258    };
259
260    let start_checkpoint = config.start_checkpoint.unwrap_or(start_checkpoint);
261    Ok(InitialState {
262        initial_head,
263        start_checkpoint,
264    })
265}

267struct PageResult {
268    events: Vec<AuthenticatedEvent>,
269    end_cursor: Option<prost::bytes::Bytes>,
270    end_reason: Option<QueryEndReason>,
271}
272
273async fn fetch_one_page(
274    light: &mut LightClient,
275    request: ListEventsRequest,
276) -> Result<PageResult, LightClientError> {
277    let mut stream = light
278        .rpc()
279        .ledger_client_alpha()
280        .list_events(request)
281        .await?
282        .into_inner();
283
284    let mut events = Vec::new();
285    let mut end_cursor = None;
286    let mut end_reason = None;
287
288    while let Some(frame) = stream.next().await {
289        let frame = frame?;
290        match frame.response {
291            Some(list_events_response::Response::Item(item)) => {
292                let ev = AuthenticatedEvent::try_from(&item)?;
293                events.push(ev);
294            }
295            Some(list_events_response::Response::End(end)) => {
296                end_cursor = end.cursor;
297                end_reason = QueryEndReason::try_from(end.reason).ok();
298                break;
299            }
300            None => break,
301        }
302    }
303
304    Ok(PageResult {
305        events,
306        end_cursor,
307        end_reason,
308    })
309}

311async fn reconcile_once(
312    light: &mut LightClient,
313    state: &mut StreamState,
314    stream_head_object_id: &sui_sdk_types::Address,
315) -> Result<Vec<AuthenticatedEvent>, LightClientError> {
316    let tip = light.latest_checkpoint_seq().await?;
317    let proof = light
318        .prove_object_at_checkpoint(stream_head_object_id, tip)
319        .await?;
320    match proof {
321        CheckpointObjectProof::Inclusion {
322            object: Some(object),
323            ..
324        } => {
325            let chain_head = extract_event_stream_head(&object)?;
326            reconcile_local_head(state, chain_head)
327        }
328        CheckpointObjectProof::Inclusion { object: None, .. } => {
329            // The on-chain head was deleted or wrapped at the tip.
330            // The local replay cannot be reconciled against a missing
331            // head; this is terminal for the stream.
332            Err(LightClientError::UnexpectedObjectShape {
333                reason: "event stream head was deleted or wrapped at the reconciliation tip",
334            })
335        }
336        // The head wasn't modified at the tip: there is no fresh
337        // on-chain head to reconcile against, so we can't confirm any
338        // additional events here. Pending events stay pending; the
339        // next reconciliation tick will retry.
340        //
341        // This is the success-path equivalent of the old code's
342        // `NotFound` error at reconcile time: today the stream would
343        // have terminated; now we let it idle until a checkpoint
344        // actually touches the head.
345        CheckpointObjectProof::NonInclusion => Ok(Vec::new()),
346    }
347}

349fn build_filter(stream_id_hex: &str) -> EventFilter {
350    EventFilter {
351        terms: vec![EventTerm {
352            literals: vec![EventLiteral {
353                polarity: Some(event_literal::Polarity::Include(EventPredicate {
354                    predicate: Some(event_predicate::Predicate::EventStreamHead(
355                        EventStreamHeadFilter {
356                            stream_id: Some(stream_id_hex.to_string()),
357                        },
358                    )),
359                })),
360            }],
361        }],
362    }
363}

365/// True if pending events were already drained up through the last
366/// folded checkpoint, so we're idle and may as well reconcile sooner.
367fn page_drain_done(state: &StreamState, next_checkpoint: u64) -> bool {
368    state.local_head.checkpoint_seq < next_checkpoint
369}

371fn e_is_retryable(err: &LightClientError) -> bool {
372    matches!(
373        err,
374        LightClientError::Rpc(status)
375            if matches!(
376                status.code(),
377                tonic::Code::Unavailable
378                    | tonic::Code::DeadlineExceeded
379                    | tonic::Code::ResourceExhausted
380                    | tonic::Code::Aborted
381            )
382    )
383}

385/// Backoff for `retry_backoff * attempts + jitter`. Returns `true` if
386/// the task should continue, `false` if `max_connect_retries` has been
387/// exhausted (in which case the error has been forwarded to the
388/// channel).
389async fn backoff_or_give_up(
390    tx: &mpsc::Sender<Result<AuthenticatedEvent, LightClientError>>,
391    config: &AuthenticatedEventsConfig,
392    consecutive_failures: &mut u32,
393    err: LightClientError,
394) -> bool {
395    *consecutive_failures += 1;
396    if *consecutive_failures > config.max_connect_retries {
397        let _ = tx.send(Err(err)).await;
398        return false;
399    }
400    let base = config.retry_backoff.saturating_mul(*consecutive_failures);
401    let jitter = pseudo_jitter(*consecutive_failures, config.retry_jitter);
402    tokio::time::sleep(base.saturating_add(jitter)).await;
403    true
404}

406/// Deterministic jitter derived from the attempt count. Avoids
407/// pulling in `rand` for a single call; the goal is just to avoid
408/// lockstep retries across many concurrent streams, not cryptographic
409/// randomness.
410fn pseudo_jitter(attempts: u32, ceiling: Duration) -> Duration {
411    if ceiling.is_zero() {
412        return Duration::ZERO;
413    }
414    // Mix the attempt count with a fixed prime to spread values.
415    let mix = (u64::from(attempts).wrapping_mul(0x9E3779B97F4A7C15)) as u128;
416    let ceiling_ms = ceiling.as_millis().max(1);
417    let offset_ms = (mix % ceiling_ms) as u64;
418    Duration::from_millis(offset_ms)
