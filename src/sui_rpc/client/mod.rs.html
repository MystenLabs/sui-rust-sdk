53    #[allow(clippy::result_large_err)]
54    pub fn new<T>(uri: T) -> Result<Self>
55    where
56        T: TryInto<http::Uri>,
57        T::Error: Into<BoxError>,
58    {
59        let uri = uri
60            .try_into()
61            .map_err(Into::into)
62            .map_err(tonic::Status::from_error)?;
63        let mut endpoint = tonic::transport::Endpoint::from(uri.clone());
64        if uri.scheme() == Some(&http::uri::Scheme::HTTPS) {
65            endpoint = endpoint
66                .tls_config(ClientTlsConfig::new().with_enabled_roots())
67                .map_err(Into::into)
68                .map_err(tonic::Status::from_error)?;
69        }
70        let channel = endpoint
71            .connect_timeout(Duration::from_secs(5))
72            .http2_keep_alive_interval(Duration::from_secs(5))
73            .connect_lazy();
74
75        Ok(Self {
76            uri,
77            channel,
78            auth: Default::default(),
79            max_decoding_message_size: None,
80        })
81    }
