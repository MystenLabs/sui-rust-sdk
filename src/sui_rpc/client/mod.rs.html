34impl Client {
35    #[allow(clippy::result_large_err)]
36    pub fn new<T>(uri: T) -> Result<Self>
37    where
38        T: TryInto<http::Uri>,
39        T::Error: Into<BoxError>,
40    {
41        let uri = uri
42            .try_into()
43            .map_err(Into::into)
44            .map_err(tonic::Status::from_error)?;
45        let mut endpoint = tonic::transport::Endpoint::from(uri.clone());
46        if uri.scheme() == Some(&http::uri::Scheme::HTTPS) {
47            endpoint = endpoint
48                .tls_config(ClientTlsConfig::new().with_enabled_roots())
49                .map_err(Into::into)
50                .map_err(tonic::Status::from_error)?;
51        }
52        let channel = endpoint
53            .connect_timeout(Duration::from_secs(5))
54            .http2_keep_alive_interval(Duration::from_secs(5))
55            .connect_lazy();
56
57        Ok(Self {
58            uri,
59            channel,
60            auth: Default::default(),
61            max_decoding_message_size: None,
62        })
63    }
